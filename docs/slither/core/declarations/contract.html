 272    @property
 273    def using_for_complete(self) -> Dict[Union[str, Type], List[Type]]:
 274        """
 275        Dict[Union[str, Type], List[Type]]: Dict of merged local using for directive with top level directive
 276        """
 277
 278        def _merge_using_for(uf1, uf2):
 279            result = {**uf1, **uf2}
 280            for key, value in result.items():
 281                if key in uf1 and key in uf2:
 282                    result[key] = value + uf1[key]
 283            return result
 284
 285        if self._using_for_complete is None:
 286            result = self.using_for
 287            top_level_using_for = self.file_scope.using_for_directives
 288            for uftl in top_level_using_for:
 289                result = _merge_using_for(result, uftl.using_for)
 290            self._using_for_complete = result
 291        return self._using_for_complete

 395    @property
 396    def state_variables_used_in_reentrant_targets(
 397        self,
 398    ) -> Dict["StateVariable", Set[Union["StateVariable", "Function"]]]:
 399        """
 400        Returns the state variables used in reentrant targets. Heuristics:
 401        - Variable used (read/write) in entry points that are reentrant
 402        - State variables that are public
 403
 404        """
 405        from slither.core.variables.state_variable import StateVariable
 406
 407        if self._state_variables_used_in_reentrant_targets is None:
 408            reentrant_functions = [f for f in self.functions_entry_points if f.is_reentrant]
 409            variables_used: Dict[
 410                StateVariable, Set[Union[StateVariable, "Function"]]
 411            ] = defaultdict(set)
 412            for function in reentrant_functions:
 413                for ir in function.all_slithir_operations():
 414                    state_variables = [v for v in ir.used if isinstance(v, StateVariable)]
 415                    for state_variable in state_variables:
 416                        variables_used[state_variable].add(ir.node.function)
 417            for variable in [v for v in self.state_variables if v.visibility == "public"]:
 418                variables_used[variable].add(variable)
 419            self._state_variables_used_in_reentrant_targets = variables_used
 420        return self._state_variables_used_in_reentrant_targets

 429    @property
 430    def constructor(self) -> Optional["Function"]:
 431        """
 432        Return the contract's immediate constructor.
 433        If there is no immediate constructor, returns the first constructor
 434        executed, following the c3 linearization
 435        Return None if there is no constructor.
 436        """
 437        cst = self.constructors_declared
 438        if cst:
 439            return cst
 440        for inherited_contract in self.inheritance:
 441            cst = inherited_contract.constructors_declared
 442            if cst:
 443                return cst
 444        return None

 625    def available_elements_from_inheritances(
 626        self,
 627        elements: Dict[str, "Function"],
 628        getter_available: Callable[["Contract"], List["FunctionContract"]],
 629    ) -> Dict[str, "Function"]:
 630        """
 631
 632        :param elements: dict(canonical_name -> elements)
 633        :param getter_available: fun x
 634        :return:
 635        """
 636        # keep track of the contracts visited
 637        # to prevent an ovveride due to multiple inheritance of the same contract
 638        # A is B, C, D is C, --> the second C was already seen
 639        inherited_elements: Dict[str, "FunctionContract"] = {}
 640        accessible_elements = {}
 641        contracts_visited = []
 642        for father in self.inheritance_reverse:
 643            functions: Dict[str, "FunctionContract"] = {
 644                v.full_name: v
 645                for v in getter_available(father)
 646                if v.contract not in contracts_visited
 647                and v.function_language
 648                != FunctionLanguage.Yul  # Yul functions are not propagated in the inheritance
 649            }
 650            contracts_visited.append(father)
 651            inherited_elements.update(functions)
 652
 653        for element in inherited_elements.values():
 654            accessible_elements[element.full_name] = elements[element.canonical_name]
 655
 656        return accessible_elements

1231    def is_upgradeable(self) -> bool:

1235                return False
1236            initializable = self.file_scope.get_contract_from_name("Initializable")
1237            if initializable:
1238                if initializable in self.inheritance:
1239                    self._is_upgradeable = True
1240            else:
1241                for contract in self.inheritance + [self]:
1242                    # This might lead to false positive
1243                    # Not sure why pylint is having a trouble here
1244                    # pylint: disable=no-member
1245                    lower_name = contract.name.lower()
1246                    if "upgradeable" in lower_name or "upgradable" in lower_name:
1247                        self._is_upgradeable = True
1248                        break
1249                    if "initializable" in lower_name:
1250                        self._is_upgradeable = True
1251                        break
1252        return self._is_upgradeable

1265            if "Proxy" in self.name:
1266                self._is_upgradeable_proxy = True

1271                        for ir in node.irs:
1272                            if isinstance(ir, LowLevelCall) and ir.function_name == "delegatecall":

1277                            if inline_asm:
1278                                if "delegatecall" in inline_asm:
1279                                    self._is_upgradeable_proxy = True

 280            result = {**uf1, **uf2}

 282                if key in uf1 and key in uf2:
 283                    result[key] = value + uf1[key]
 284            return result

 289            for uftl in top_level_using_for:
 290                result = _merge_using_for(result, uftl.using_for)
 291            self._using_for_complete = result

 626    def available_elements_from_inheritances(
 627        self,
 628        elements: Dict[str, "Function"],
 629        getter_available: Callable[["Contract"], List["FunctionContract"]],
 630    ) -> Dict[str, "Function"]:
 631        """
 632
 633        :param elements: dict(canonical_name -> elements)
 634        :param getter_available: fun x
 635        :return:
 636        """
 637        # keep track of the contracts visited
 638        # to prevent an ovveride due to multiple inheritance of the same contract
 639        # A is B, C, D is C, --> the second C was already seen
 640        inherited_elements: Dict[str, "FunctionContract"] = {}
 641        accessible_elements = {}
 642        contracts_visited = []
 643        for father in self.inheritance_reverse:
 644            functions: Dict[str, "FunctionContract"] = {
 645                v.full_name: v
 646                for v in getter_available(father)
 647                if v.contract not in contracts_visited
 648                and v.function_language
 649                != FunctionLanguage.Yul  # Yul functions are not propagated in the inheritance
 650            }
 651            contracts_visited.append(father)
 652            inherited_elements.update(functions)
 653
 654        for element in inherited_elements.values():
 655            accessible_elements[element.full_name] = elements[element.canonical_name]
 656
 657        return accessible_elements
