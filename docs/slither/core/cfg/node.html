  71    # Node with expression

  76    VARIABLE = 0x13  # Declaration of variable

  78    IFLOOP = 0x15

 464    def can_reenter(self, callstack=None) -> bool:
 465        """
 466        Check if the node can re-enter
 467        Do not consider CREATE as potential re-enter, but check if the
 468        destination's constructor can contain a call (recurs. follow nested CREATE)
 469        For Solidity > 0.5, filter access to public variables and constant/pure/view
 470        For call to this. check if the destination can re-enter
 471        Do not consider Send/Transfer as there is not enough gas
 472        :param callstack: used internally to check for recursion
 473        :return bool:
 474        """
 475        # pylint: disable=import-outside-toplevel
 476        from slither.slithir.operations import Call
 477
 478        if self._can_reenter is None:
 479            self._can_reenter = False
 480            for ir in self.irs:
 481                if isinstance(ir, Call) and ir.can_reenter(callstack):
 482                    self._can_reenter = True
 483                    return True
 484        return self._can_reenter
 485
 486    def can_send_eth(self) -> bool:
 487        """
 488        Check if the node can send eth
 489        :return bool:
 490        """
 491        # pylint: disable=import-outside-toplevel
 492        from slither.slithir.operations import Call
 493
 494        if self._can_send_eth is None:
 495            self._can_send_eth = False
 496            for ir in self.all_slithir_operations():
 497                if isinstance(ir, Call) and ir.can_send_eth():
 498                    self._can_send_eth = True
 499                    return True
 500        return self._can_send_eth

 553    def contains_if(self, include_loop=True) -> bool:
 554        """
 555            Check if the node is a IF node
 556        Returns:
 557            bool: True if the node is a conditional node (IF or IFLOOP)
 558        """
 559        if include_loop:
 560            return self.type in [NodeType.IF, NodeType.IFLOOP]
 561        return self.type == NodeType.IF
 562
 563    def is_conditional(self, include_loop=True) -> bool:
 564        """
 565            Check if the node is a conditional node
 566            A conditional node is either a IF or a require/assert or a RETURN bool
 567        Returns:
 568            bool: True if the node is a conditional node
 569        """
 570        if self.contains_if(include_loop) or self.contains_require_or_assert():
 571            return True
 572        if self.irs:
 573            last_ir = self.irs[-1]
 574            if last_ir:
 575                if isinstance(last_ir, Return):
 576                    for r in last_ir.read:
 577                        if r.type == ElementaryType("bool"):
 578                            return True
 579        return False

 465    def can_reenter(self, callstack=None) -> bool:
 466        """
 467        Check if the node can re-enter
 468        Do not consider CREATE as potential re-enter, but check if the
 469        destination's constructor can contain a call (recurs. follow nested CREATE)
 470        For Solidity > 0.5, filter access to public variables and constant/pure/view
 471        For call to this. check if the destination can re-enter
 472        Do not consider Send/Transfer as there is not enough gas
 473        :param callstack: used internally to check for recursion
 474        :return bool:
 475        """
 476        # pylint: disable=import-outside-toplevel
 477        from slither.slithir.operations import Call
 478
 479        if self._can_reenter is None:
 480            self._can_reenter = False
 481            for ir in self.irs:
 482                if isinstance(ir, Call) and ir.can_reenter(callstack):
 483                    self._can_reenter = True
 484                    return True
 485        return self._can_reenter

 565        """
 566            Check if the node is a conditional node

 568        Returns:
 569            bool: True if the node is a conditional node
 570        """
 571        if self.contains_if(include_loop) or self.contains_require_or_assert():

 576                if isinstance(last_ir, Return):
 577                    for r in last_ir.read:
 578                        if r.type == ElementaryType("bool"):
 579                            return True

 862    def _find_read_write_call(self):  # pylint: disable=too-many-statements
 863
 864        for ir in self.irs:
 865
 866            self._slithir_vars |= {v for v in ir.read if self._is_valid_slithir_var(v)}
 867
 868            if isinstance(ir, OperationWithLValue):
 869                var = ir.lvalue
 870                if var and self._is_valid_slithir_var(var):
 871                    self._slithir_vars.add(var)
 872
 873            if not isinstance(ir, (Phi, Index, Member)):
 874                self._vars_read += [v for v in ir.read if self._is_non_slithir_var(v)]
 875                for var in ir.read:
 876                    if isinstance(var, ReferenceVariable):
 877                        self._vars_read.append(var.points_to_origin)
 878            elif isinstance(ir, (Member, Index)):
 879                var = ir.variable_left if isinstance(ir, Member) else ir.variable_right
 880                if self._is_non_slithir_var(var):
 881                    self._vars_read.append(var)
 882                if isinstance(var, ReferenceVariable):
 883                    origin = var.points_to_origin
 884                    if self._is_non_slithir_var(origin):
 885                        self._vars_read.append(origin)
 886
 887            if isinstance(ir, OperationWithLValue):
 888                if isinstance(ir, (Index, Member, Length)):
 889                    continue  # Don't consider Member and Index operations -> ReferenceVariable
 890                var = ir.lvalue
 891                if isinstance(var, ReferenceVariable):
 892                    var = var.points_to_origin
 893                if var and self._is_non_slithir_var(var):
 894                    self._vars_written.append(var)
 895
 896            if isinstance(ir, InternalCall):
 897                self._internal_calls.append(ir.function)
 898            if isinstance(ir, SolidityCall):
 899                # TODO: consider removing dependancy of solidity_call to internal_call
 900                self._solidity_calls.append(ir.function)
 901                self._internal_calls.append(ir.function)
 902            if isinstance(ir, LowLevelCall):
 903                assert isinstance(ir.destination, (Variable, SolidityVariable))
 904                self._low_level_calls.append((ir.destination, ir.function_name.value))
 905            elif isinstance(ir, HighLevelCall) and not isinstance(ir, LibraryCall):
 906                if isinstance(ir.destination.type, Contract):
 907                    self._high_level_calls.append((ir.destination.type, ir.function))
 908                elif ir.destination == SolidityVariable("this"):
 909                    self._high_level_calls.append((self.function.contract, ir.function))
 910                else:
 911                    try:
 912                        self._high_level_calls.append((ir.destination.type.type, ir.function))
 913                    except AttributeError as error:
 914                        #  pylint: disable=raise-missing-from
 915                        raise SlitherException(
 916                            f"Function not found on IR: {ir}.\nNode: {self} ({self.source_mapping})\nFunction: {self.function}\nPlease try compiling with a recent Solidity version. {error}"
 917                        )
 918            elif isinstance(ir, LibraryCall):
 919                assert isinstance(ir.destination, Contract)
 920                assert isinstance(ir.function, Function)
 921                self._high_level_calls.append((ir.destination, ir.function))
 922                self._library_calls.append((ir.destination, ir.function))
 923
 924        self._vars_read = list(set(self._vars_read))
 925        self._state_vars_read = [v for v in self._vars_read if isinstance(v, StateVariable)]
 926        self._local_vars_read = [v for v in self._vars_read if isinstance(v, LocalVariable)]
 927        self._solidity_vars_read = [v for v in self._vars_read if isinstance(v, SolidityVariable)]
 928        self._vars_written = list(set(self._vars_written))
 929        self._state_vars_written = [v for v in self._vars_written if isinstance(v, StateVariable)]
 930        self._local_vars_written = [v for v in self._vars_written if isinstance(v, LocalVariable)]
 931        self._internal_calls = list(set(self._internal_calls))
 932        self._solidity_calls = list(set(self._solidity_calls))
 933        self._high_level_calls = list(set(self._high_level_calls))
 934        self._library_calls = list(set(self._library_calls))
 935        self._low_level_calls = list(set(self._low_level_calls))
 936

 938    def _convert_ssa(v: Variable):
 939        if isinstance(v, StateIRVariable):
 940            contract = v.contract
 941            non_ssa_var = contract.get_state_variable_from_name(v.name)
 942            return non_ssa_var

 948    def update_read_write_using_ssa(self):
 949        if not self.expression:
 950            return
 951        for ir in self.irs_ssa:
 952            if isinstance(ir, PhiCallback):
 953                continue
 954            if not isinstance(ir, (Phi, Index, Member)):
 955                self._ssa_vars_read += [
 956                    v for v in ir.read if isinstance(v, (StateIRVariable, LocalIRVariable))
 957                ]
 958                for var in ir.read:
 959                    if isinstance(var, ReferenceVariable):
 960                        origin = var.points_to_origin
 961                        if isinstance(origin, (StateIRVariable, LocalIRVariable)):
 962                            self._ssa_vars_read.append(origin)
 963
 964            elif isinstance(ir, (Member, Index)):
 965                if isinstance(ir.variable_right, (StateIRVariable, LocalIRVariable)):
 966                    self._ssa_vars_read.append(ir.variable_right)
 967                if isinstance(ir.variable_right, ReferenceVariable):
 968                    origin = ir.variable_right.points_to_origin
 969                    if isinstance(origin, (StateIRVariable, LocalIRVariable)):
 970                        self._ssa_vars_read.append(origin)
 971
 972            if isinstance(ir, OperationWithLValue):
 973                if isinstance(ir, (Index, Member, Length)):
 974                    continue  # Don't consider Member and Index operations -> ReferenceVariable
 975                var = ir.lvalue
 976                if isinstance(var, ReferenceVariable):
 977                    var = var.points_to_origin
 978                # Only store non-slithIR variables
 979                if var and isinstance(var, (StateIRVariable, LocalIRVariable)):
 980                    if isinstance(ir, PhiCallback):
 981                        continue
 982                    self._ssa_vars_written.append(var)
