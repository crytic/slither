196    def _compute_offsets_from_thing(self, thing: SourceMapping):
197        definition = get_definition(thing, self.crytic_compile)
198        references = get_references(thing)
199        implementation = get_implementation(thing)
200
201        for offset in range(definition.start, definition.end + 1):
202
203            if (
204                isinstance(thing, TopLevel)
205                or (
206                    isinstance(thing, FunctionContract)
207                    and thing.contract_declarer == thing.contract
208                )
209                or (isinstance(thing, ChildContract) and not isinstance(thing, FunctionContract))
210            ):
211                self._offset_to_objects[definition.filename][offset].add(thing)
212
213            self._offset_to_definitions[definition.filename][offset].add(definition)
214            self._offset_to_implementations[definition.filename][offset].add(implementation)
215            self._offset_to_references[definition.filename][offset] |= set(references)
216
217        for ref in references:
218            for offset in range(ref.start, ref.end + 1):
219
220                if (
221                    isinstance(thing, TopLevel)
222                    or (
223                        isinstance(thing, FunctionContract)
224                        and thing.contract_declarer == thing.contract
225                    )
226                    or (
227                        isinstance(thing, ChildContract) and not isinstance(thing, FunctionContract)
228                    )
229                ):
230                    self._offset_to_objects[definition.filename][offset].add(thing)
231
232                self._offset_to_definitions[ref.filename][offset].add(definition)
233                self._offset_to_implementations[ref.filename][offset].add(implementation)

236    def _compute_offsets_to_ref_impl_decl(self):  # pylint: disable=too-many-branches
237        self._offset_to_references = defaultdict(lambda: defaultdict(lambda: set()))
238        self._offset_to_definitions = defaultdict(lambda: defaultdict(lambda: set()))
239        self._offset_to_implementations = defaultdict(lambda: defaultdict(lambda: set()))
240        self._offset_to_objects = defaultdict(lambda: defaultdict(lambda: set()))
241
242        for compilation_unit in self._compilation_units:
243            for contract in compilation_unit.contracts:
244                self._compute_offsets_from_thing(contract)
245
246                for function in contract.functions:
247                    self._compute_offsets_from_thing(function)
248                    for variable in function.local_variables:
249                        self._compute_offsets_from_thing(variable)
250                for modifier in contract.modifiers:
251                    self._compute_offsets_from_thing(modifier)
252                    for variable in modifier.local_variables:
253                        self._compute_offsets_from_thing(variable)
254
255                for st in contract.structures:
256                    self._compute_offsets_from_thing(st)
257
258                for enum in contract.enums:
259                    self._compute_offsets_from_thing(enum)
260
261                for event in contract.events:
262                    self._compute_offsets_from_thing(event)
263            for enum in compilation_unit.enums_top_level:
264                self._compute_offsets_from_thing(enum)
265            for function in compilation_unit.functions_top_level:
266                self._compute_offsets_from_thing(function)
267            for st in compilation_unit.structures_top_level:
268                self._compute_offsets_from_thing(st)
269            for import_directive in compilation_unit.import_directives:
270                self._compute_offsets_from_thing(import_directive)
271            for pragma in compilation_unit.pragma_directives:

299    def parse_ignore_comments(self, file: str) -> None:
300        # The first time we check a file, find all start/end ignore comments and memoize them.
301        line_number = 1
302        while True:
303
304            line_text = self.crytic_compile.get_code_from_line(file, line_number)
305            if line_text is None:
306                break
307
308            start_regex = r"^\s*//\s*slither-disable-start\s*([a-zA-Z0-9_,-]*)"
309            end_regex = r"^\s*//\s*slither-disable-end\s*([a-zA-Z0-9_,-]*)"
310            start_match = re.findall(start_regex, line_text.decode("utf8"))
311            end_match = re.findall(end_regex, line_text.decode("utf8"))
312
313            if start_match:
314                ignored = start_match[0].split(",")
315                if ignored:
316                    for check in ignored:
317                        vals = self._ignore_ranges[file][check]
318                        if len(vals) == 0 or vals[-1][1] != float("inf"):
319                            # First item in the array, or the prior item is fully populated.
320                            self._ignore_ranges[file][check].append((line_number, float("inf")))
321                        else:
322                            logger.error(
323                                f"Consecutive slither-disable-starts without slither-disable-end in {file}#{line_number}"
324                            )
325                            return
326
327            if end_match:
328                ignored = end_match[0].split(",")
329                if ignored:
330                    for check in ignored:
331                        vals = self._ignore_ranges[file][check]
332                        if len(vals) == 0 or vals[-1][1] != float("inf"):
333                            logger.error(
334                                f"slither-disable-end without slither-disable-start in {file}#{line_number}"
335                            )
336                            return
337                        self._ignore_ranges[file][check][-1] = (vals[-1][0], line_number)
338
339            line_number += 1

341    def has_ignore_comment(self, r: Dict) -> bool:
342        """
343        Check if the result has an ignore comment in the file or on the preceding line, in which
344        case, it is not valid
345        """
346        if not self.crytic_compile:
347            return False
348        mapping_elements_with_lines = (
349            (
350                posixpath.normpath(elem["source_mapping"]["filename_absolute"]),
351                elem["source_mapping"]["lines"],
352            )
353            for elem in r["elements"]
354            if "source_mapping" in elem
355            and "filename_absolute" in elem["source_mapping"]
356            and "lines" in elem["source_mapping"]
357            and len(elem["source_mapping"]["lines"]) > 0
358        )
359
360        for file, lines in mapping_elements_with_lines:
361
362            # Check if result is within an ignored range.
363            ignore_ranges = self._ignore_ranges[file][r["check"]] + self._ignore_ranges[file]["all"]
364            for start, end in ignore_ranges:
365                # The full check must be within the ignore range to be ignored.
366                if start < lines[0] and end > lines[-1]:
367                    return True
368
369            # Check for next-line matchers.
370            ignore_line_index = min(lines) - 1
371            ignore_line_text = self.crytic_compile.get_code_from_line(file, ignore_line_index)
372            if ignore_line_text:
373                match = re.findall(
374                    r"^\s*//\s*slither-disable-next-line\s*([a-zA-Z0-9_,-]*)",
375                    ignore_line_text.decode("utf8"),
376                )
377                if match:
378                    ignored = match[0].split(",")
379                    if ignored and ("all" in ignored or any(r["check"] == c for c in ignored)):
380                        return True
381
382        return False

384    def valid_result(self, r: Dict) -> bool:
385        """
386        Check if the result is valid
387        A result is invalid if:
388            - All its source paths belong to the source path filtered
389            - Or a similar result was reported and saved during a previous run
390            - The --exclude-dependencies flag is set and results are only related to dependencies
391            - There is an ignore comment on the preceding line or in the file
392        """
393
394        # Remove duplicate due to the multiple compilation support
395        if r["id"] in self._currently_seen_resuts:
396            return False
397        self._currently_seen_resuts.add(r["id"])
398
399        source_mapping_elements = [
400            elem["source_mapping"].get("filename_absolute", "unknown")
401            for elem in r["elements"]
402            if "source_mapping" in elem
403        ]
404
405        # Use POSIX-style paths so that filter_paths works across different
406        # OSes. Convert to a list so elements don't get consumed and are lost
407        # while evaluating the first pattern
408        source_mapping_elements = list(
409            map(lambda x: pathlib.Path(x).resolve().as_posix() if x else x, source_mapping_elements)
410        )
411        matching = False
412
413        for path in self._paths_to_filter:
414            try:
415                if any(
416                    bool(re.search(_relative_path_format(path), src_mapping))
417                    for src_mapping in source_mapping_elements
418                ):
419                    matching = True
420                    break
421            except re.error:
422                logger.error(
423                    f"Incorrect regular expression for --filter-paths {path}."
424                    "\nSlither supports the Python re format"
425                    ": https://docs.python.org/3/library/re.html"
426                )
427
428        if r["elements"] and matching:
429            return False
430
431        if self._show_ignored_findings:
432            return True
433        if self.has_ignore_comment(r):
434            return False
435        if r["id"] in self._previous_results_ids:
436            return False
437        if r["elements"] and self._exclude_dependencies:
438            if all(element["source_mapping"]["is_dependency"] for element in r["elements"]):
439                return False
440        # Conserve previous result filtering. This is conserved for compatibility, but is meant to be removed
441        if r["description"] in [pr["description"] for pr in self._previous_results]:
442            return False
443
444        return True

446    def load_previous_results(self):
447        filename = self._previous_results_filename
448        try:
449            if os.path.isfile(filename):
450                with open(filename, encoding="utf8") as f:
451                    self._previous_results = json.load(f)
452                    if self._previous_results:
453                        for r in self._previous_results:
454                            if "id" in r:
455                                self._previous_results_ids.add(r["id"])
456        except json.decoder.JSONDecodeError:
457            logger.error(red(f"Impossible to decode {filename}. Consider removing the file"))
458

464            results = self._results_to_hide + self._previous_results

300    def parse_ignore_comments(self, file: str) -> None:
301        # The first time we check a file, find all start/end ignore comments and memoize them.
302        line_number = 1
303        while True:
304
305            line_text = self.crytic_compile.get_code_from_line(file, line_number)
306            if line_text is None:
307                break
308
309            start_regex = r"^\s*//\s*slither-disable-start\s*([a-zA-Z0-9_,-]*)"
310            end_regex = r"^\s*//\s*slither-disable-end\s*([a-zA-Z0-9_,-]*)"
311            start_match = re.findall(start_regex, line_text.decode("utf8"))
312            end_match = re.findall(end_regex, line_text.decode("utf8"))
313
314            if start_match:
315                ignored = start_match[0].split(",")
316                if ignored:
317                    for check in ignored:
318                        vals = self._ignore_ranges[file][check]
319                        if len(vals) == 0 or vals[-1][1] != float("inf"):
320                            # First item in the array, or the prior item is fully populated.
321                            self._ignore_ranges[file][check].append((line_number, float("inf")))
322                        else:
323                            logger.error(
324                                f"Consecutive slither-disable-starts without slither-disable-end in {file}#{line_number}"
325                            )
326                            return
327
328            if end_match:
329                ignored = end_match[0].split(",")
330                if ignored:
331                    for check in ignored:
332                        vals = self._ignore_ranges[file][check]
333                        if len(vals) == 0 or vals[-1][1] != float("inf"):
334                            logger.error(
335                                f"slither-disable-end without slither-disable-start in {file}#{line_number}"
336                            )
337                            return
338                        self._ignore_ranges[file][check][-1] = (vals[-1][0], line_number)
339
340            line_number += 1

342    def has_ignore_comment(self, r: Dict) -> bool:
343        """
344        Check if the result has an ignore comment in the file or on the preceding line, in which
345        case, it is not valid
346        """
347        if not self.crytic_compile:
348            return False
349        mapping_elements_with_lines = (
350            (
351                posixpath.normpath(elem["source_mapping"]["filename_absolute"]),
352                elem["source_mapping"]["lines"],
353            )
354            for elem in r["elements"]
355            if "source_mapping" in elem
356            and "filename_absolute" in elem["source_mapping"]
357            and "lines" in elem["source_mapping"]
358            and len(elem["source_mapping"]["lines"]) > 0
359        )
360
361        for file, lines in mapping_elements_with_lines:
362
363            # Check if result is within an ignored range.
364            ignore_ranges = self._ignore_ranges[file][r["check"]] + self._ignore_ranges[file]["all"]
365            for start, end in ignore_ranges:
366                # The full check must be within the ignore range to be ignored.
367                if start < lines[0] and end > lines[-1]:
368                    return True
369
370            # Check for next-line matchers.
371            ignore_line_index = min(lines) - 1
372            ignore_line_text = self.crytic_compile.get_code_from_line(file, ignore_line_index)
373            if ignore_line_text:
374                match = re.findall(
375                    r"^\s*//\s*slither-disable-next-line\s*([a-zA-Z0-9_,-]*)",
376                    ignore_line_text.decode("utf8"),
377                )
378                if match:
379                    ignored = match[0].split(",")
380                    if ignored and ("all" in ignored or any(r["check"] == c for c in ignored)):
381                        return True
382
383        return False

385    def valid_result(self, r: Dict) -> bool:
386        """
387        Check if the result is valid
388        A result is invalid if:
389            - All its source paths belong to the source path filtered
390            - Or a similar result was reported and saved during a previous run
391            - The --exclude-dependencies flag is set and results are only related to dependencies
392            - There is an ignore comment on the preceding line or in the file
393        """
394
395        # Remove duplicate due to the multiple compilation support
396        if r["id"] in self._currently_seen_resuts:
397            return False
398        self._currently_seen_resuts.add(r["id"])
399
400        source_mapping_elements = [
401            elem["source_mapping"].get("filename_absolute", "unknown")
402            for elem in r["elements"]
403            if "source_mapping" in elem
404        ]
405
406        # Use POSIX-style paths so that filter_paths works across different
407        # OSes. Convert to a list so elements don't get consumed and are lost
408        # while evaluating the first pattern
409        source_mapping_elements = list(
410            map(lambda x: pathlib.Path(x).resolve().as_posix() if x else x, source_mapping_elements)
411        )
412        matching = False
413
414        for path in self._paths_to_filter:
415            try:
416                if any(
417                    bool(re.search(_relative_path_format(path), src_mapping))
418                    for src_mapping in source_mapping_elements
419                ):
420                    matching = True
421                    break
422            except re.error:
423                logger.error(
424                    f"Incorrect regular expression for --filter-paths {path}."
425                    "\nSlither supports the Python re format"
426                    ": https://docs.python.org/3/library/re.html"
427                )
428
429        if r["elements"] and matching:
430            return False
431
432        if self._show_ignored_findings:
433            return True
434        if self.has_ignore_comment(r):
435            return False
436        if r["id"] in self._previous_results_ids:
437            return False
438        if r["elements"] and self._exclude_dependencies:
439            if all(element["source_mapping"]["is_dependency"] for element in r["elements"]):
440                return False
441        # Conserve previous result filtering. This is conserved for compatibility, but is meant to be removed
442        if r["description"] in [pr["description"] for pr in self._previous_results]:
443            return False
444
445        return True
