  94def convert_expression(expression, node):
  95    # handle standlone expression
  96    # such as return true;
  97    from slither.core.cfg.node import NodeType
  98
  99    if isinstance(expression, Literal) and node.type in [NodeType.IF, NodeType.IFLOOP]:
 100        cst = Constant(expression.value, expression.type)
 101        cond = Condition(cst)
 102        cond.set_expression(expression)
 103        cond.set_node(node)
 104        result = [cond]
 105        return result
 106    if isinstance(expression, Identifier) and node.type in [
 107        NodeType.IF,
 108        NodeType.IFLOOP,
 109    ]:
 110        cond = Condition(expression.value)
 111        cond.set_expression(expression)
 112        cond.set_node(node)
 113        result = [cond]
 114        return result
 115
 116    visitor = ExpressionToSlithIR(expression, node)
 117    result = visitor.result()
 118
 119    result = apply_ir_heuristics(result, node)
 120
 121    if result:
 122        if node.type in [NodeType.IF, NodeType.IFLOOP]:
 123            assert isinstance(result[-1], (OperationWithLValue))
 124            cond = Condition(result[-1].lvalue)
 125            cond.set_expression(expression)
 126            cond.set_node(node)
 127            result.append(cond)
 128        elif node.type == NodeType.RETURN:
 129            # May return None
 130            if isinstance(result[-1], (OperationWithLValue)):
 131                r = Return(result[-1].lvalue)
 132                r.set_expression(expression)
 133                r.set_node(node)
 134                result.append(r)
 135
 136    return result

 162def _fits_under_integer(val: int, can_be_int: bool, can_be_uint) -> List[str]:
 163    """
 164    Return the list of uint/int that can contain val
 165
 166    :param val:
 167    :return:
 168    """
 169    ret: List[str] = []
 170    n = 8
 171    assert can_be_int | can_be_uint
 172    while n <= 256:
 173        if can_be_uint:
 174            if val <= 2**n - 1:
 175                ret.append(f"uint{n}")
 176        if can_be_int:
 177            if val <= (2**n) / 2 - 1:
 178                ret.append(f"int{n}")
 179        n = n + 8
 180    return ret

 183def _fits_under_byte(val: Union[int, str]) -> List[str]:
 184    """
 185    Return the list of byte that can contain val
 186
 187    :param val:
 188    :return:
 189    """
 190
 191    # If the value is written as an int, it can only be saved in one byte size
 192    # If its a string, it can be fitted under multiple values
 193
 194    if isinstance(val, int):
 195        hex_val = hex(val)[2:]
 196        size = len(hex_val) // 2
 197        return [f"bytes{size}"]
 198    # val is a str
 199    length = len(val.encode("utf-8"))
 200    return [f"bytes{f}" for f in range(length, 33)] + ["bytes"]

 203def _find_function_from_parameter(
 204    arguments: List[Variable], candidates: List[Function], full_comparison: bool
 205) -> Optional[Function]:
 206    """
 207    Look for a function in candidates that can be the target based on the ir's call arguments
 208
 209    Try the implicit type conversion for uint/int/bytes. Constant values can be both uint/int
 210    While variables stick to their base type, but can changed the size.
 211    If full_comparison is True it will do a comparison of all the arguments regardless if
 212    the candidate remained is one.
 213
 214    :param arguments:
 215    :param candidates:
 216    :param full_comparison:
 217    :return:
 218    """
 219    type_args: List[str]
 220    for idx, arg in enumerate(arguments):
 221        if isinstance(arg, (list,)):
 222            type_args = [f"{get_type(arg[0].type)}[{len(arg)}]"]
 223        elif isinstance(arg, Function):
 224            type_args = [arg.signature_str]
 225        else:
 226            type_args = [get_type(arg.type)]
 227
 228        arg_type = arg.type
 229        if isinstance(
 230            arg_type, ElementaryType
 231        ) and arg_type.type in ElementaryTypeInt + Uint + Byte + ["string"]:
 232            if isinstance(arg, Constant):
 233                value = arg.value
 234                can_be_uint = True
 235                can_be_int = True
 236            else:
 237                value = MaxValues[arg_type.type]
 238                can_be_uint = False
 239                can_be_int = False
 240                if arg_type.type in ElementaryTypeInt:
 241                    can_be_int = True
 242                elif arg_type.type in Uint:
 243                    can_be_uint = True
 244
 245            if arg_type.type in ElementaryTypeInt + Uint:
 246                type_args = _fits_under_integer(value, can_be_int, can_be_uint)
 247            elif value is None and arg_type.type in ["bytes", "string"]:
 248                type_args = ["bytes", "string"]
 249            else:
 250                type_args = _fits_under_byte(value)
 251                if arg_type.type == "string":
 252                    type_args += ["string"]
 253
 254        not_found = True
 255        candidates_kept = []
 256        for type_arg in type_args:
 257            if not not_found:
 258                break
 259            candidates_kept = []
 260            for candidate in candidates:
 261                param = get_type(candidate.parameters[idx].type)
 262                if param == type_arg:
 263                    not_found = False
 264                    candidates_kept.append(candidate)
 265
 266            if len(candidates_kept) == 1 and not full_comparison:
 267                return candidates_kept[0]
 268        candidates = candidates_kept
 269    if len(candidates) == 1:
 270        return candidates[0]
 271    return None

 303def integrate_value_gas(result):
 304    """
 305    Integrate value and gas temporary arguments to call instruction
 306    """
 307    was_changed = True
 308
 309    calls = []
 310
 311    while was_changed:
 312        # We loop until we do not find any call to value or gas
 313        was_changed = False
 314
 315        # Find all the assignments
 316        assigments = {}
 317        for i in result:
 318            if isinstance(i, OperationWithLValue):
 319                assigments[i.lvalue.name] = i
 320            if isinstance(i, TmpCall):
 321                if isinstance(i.called, Variable) and i.called.name in assigments:
 322                    ins_ori = assigments[i.called.name]
 323                    i.set_ori(ins_ori)
 324
 325        to_remove = []
 326        variable_to_replace = {}
 327
 328        # Replace call to value, gas to an argument of the real call
 329        for idx, ins in enumerate(result):
 330            # value can be shadowed, so we check that the prev ins
 331            # is an Argument
 332            if is_value(ins) and isinstance(result[idx - 1], Argument):
 333                was_changed = True
 334                result[idx - 1].set_type(ArgumentType.VALUE)
 335                result[idx - 1].call_id = ins.ori.variable_left.name
 336                calls.append(ins.ori.variable_left)
 337                to_remove.append(ins)
 338                variable_to_replace[ins.lvalue.name] = ins.ori.variable_left
 339            elif is_gas(ins) and isinstance(result[idx - 1], Argument):
 340                was_changed = True
 341                result[idx - 1].set_type(ArgumentType.GAS)
 342                result[idx - 1].call_id = ins.ori.variable_left.name
 343                calls.append(ins.ori.variable_left)
 344                to_remove.append(ins)
 345                variable_to_replace[ins.lvalue.name] = ins.ori.variable_left
 346
 347        # Remove the call to value/gas instruction
 348        result = [i for i in result if not i in to_remove]
 349
 350        # update the real call
 351        for ins in result:
 352            if isinstance(ins, TmpCall):
 353                # use of while if there redirections
 354                while ins.called.name in variable_to_replace:
 355                    was_changed = True
 356                    ins.call_id = variable_to_replace[ins.called.name].name
 357                    calls.append(ins.called)
 358                    ins.called = variable_to_replace[ins.called.name]
 359            if isinstance(ins, Argument):
 360                while ins.call_id in variable_to_replace:
 361                    was_changed = True
 362                    ins.call_id = variable_to_replace[ins.call_id].name
 363
 364    calls = list({str(c) for c in calls})
 365    idx = 0
 366    calls_d = {}
 367    for call in calls:
 368        calls_d[str(call)] = idx
 369        idx = idx + 1
 370
 371    return result

 382def propagate_type_and_convert_call(result: List[Operation], node: "Node") -> List[Operation]:
 383    """
 384    Propagate the types variables and convert tmp call to real call operation
 385    """
 386    calls_value = {}
 387    calls_gas = {}
 388
 389    call_data = []
 390
 391    idx = 0
 392    # use of while len() as result can be modified during the iteration
 393    while idx < len(result):
 394        ins = result[idx]
 395
 396        if isinstance(ins, TmpCall):
 397            # If the node.function is a FunctionTopLevel, then it does not have a contract
 398            contract = (
 399                node.function.contract if isinstance(node.function, FunctionContract) else None
 400            )
 401            new_ins = extract_tmp_call(ins, contract)
 402            if new_ins:
 403                new_ins.set_node(ins.node)
 404                ins = new_ins
 405                result[idx] = ins
 406
 407        # If there's two consecutive type conversions, keep only the last
 408        # ARG_CALL x                             call_data = [x]
 409        # TMP_4 = CONVERT x to Fix               call_data = []
 410        # TMP_5 = CONVERT TMP_4 to uint192       call_data = [TMP_5]
 411        if isinstance(ins, TypeConversion) and ins.variable in call_data:
 412            call_data.remove(ins.variable)
 413
 414        if isinstance(ins, Argument):
 415            # In case of dupplicate arguments we overwrite the value
 416            # This can happen because of addr.call.value(1).value(2)
 417            if ins.get_type() in [ArgumentType.GAS]:
 418                calls_gas[ins.call_id] = ins.argument
 419            elif ins.get_type() in [ArgumentType.VALUE]:
 420                calls_value[ins.call_id] = ins.argument
 421            else:
 422                assert ins.get_type() == ArgumentType.CALL
 423                call_data.append(ins.argument)
 424
 425        if isinstance(ins, (HighLevelCall, NewContract, InternalDynamicCall)):
 426            if ins.call_id in calls_value:
 427                ins.call_value = calls_value[ins.call_id]
 428            if ins.call_id in calls_gas:
 429                ins.call_gas = calls_gas[ins.call_id]
 430
 431        if isinstance(ins, (Call, NewContract, NewStructure)):
 432            # We might have stored some arguments for libraries
 433            if ins.arguments:
 434                call_data = ins.arguments + call_data
 435            ins.arguments = call_data
 436            call_data = []
 437
 438        if is_temporary(ins):
 439            del result[idx]
 440            continue
 441
 442        new_ins = propagate_types(ins, node)
 443        if new_ins:
 444            if isinstance(new_ins, (list,)):
 445                for new_ins_ in new_ins:
 446                    new_ins_.set_node(ins.node)
 447                del result[idx]
 448                for i, ins in enumerate(new_ins):
 449                    result.insert(idx + i, ins)
 450                idx = idx + len(new_ins) - 1
 451            else:
 452                new_ins.set_node(ins.node)
 453                result[idx] = new_ins
 454        idx = idx + 1
 455    return result

 458def _convert_type_contract(ir: Member) -> Assignment:
 459    assert isinstance(ir.variable_left.type, TypeInformation)
 460    contract = ir.variable_left.type.type
 461
 462    scope = ir.node.file_scope
 463
 464    if ir.variable_right == "creationCode":
 465        bytecode = scope.bytecode_init(
 466            ir.node.compilation_unit.crytic_compile_compilation_unit, contract.name
 467        )
 468        assignment = Assignment(ir.lvalue, Constant(str(bytecode)), ElementaryType("bytes"))
 469        assignment.set_expression(ir.expression)
 470        assignment.set_node(ir.node)
 471        assignment.lvalue.set_type(ElementaryType("bytes"))
 472        return assignment
 473    if ir.variable_right == "runtimeCode":
 474        bytecode = scope.bytecode_runtime(
 475            ir.node.compilation_unit.crytic_compile_compilation_unit, contract.name
 476        )
 477        assignment = Assignment(ir.lvalue, Constant(str(bytecode)), ElementaryType("bytes"))
 478        assignment.set_expression(ir.expression)
 479        assignment.set_node(ir.node)
 480        assignment.lvalue.set_type(ElementaryType("bytes"))
 481        return assignment
 482    if ir.variable_right == "interfaceId":
 483        entry_points = contract.functions_entry_points
 484        interfaceId = 0
 485        for entry_point in entry_points:
 486            interfaceId = interfaceId ^ get_function_id(entry_point.full_name)
 487        assignment = Assignment(
 488            ir.lvalue,
 489            Constant(str(interfaceId), constant_type=ElementaryType("bytes4")),
 490            ElementaryType("bytes4"),
 491        )
 492        assignment.set_expression(ir.expression)
 493        assignment.set_node(ir.node)
 494        assignment.lvalue.set_type(ElementaryType("bytes4"))
 495        return assignment
 496
 497    if ir.variable_right == "name":
 498        assignment = Assignment(ir.lvalue, Constant(contract.name), ElementaryType("string"))
 499        assignment.set_expression(ir.expression)
 500        assignment.set_node(ir.node)
 501        assignment.lvalue.set_type(ElementaryType("string"))
 502        return assignment
 503
 504    raise SlithIRError(f"type({contract.name}).{ir.variable_right} is unknown")

 507def propagate_types(ir, node: "Node"):  # pylint: disable=too-many-locals
 508    # propagate the type
 509    node_function = node.function
 510    using_for = (
 511        node_function.contract.using_for_complete
 512        if isinstance(node_function, FunctionContract)
 513        else {}
 514    )
 515    if isinstance(ir, OperationWithLValue):
 516        # Force assignment in case of missing previous correct type
 517        if not ir.lvalue.type:
 518            if isinstance(ir, Assignment):
 519                ir.lvalue.set_type(ir.rvalue.type)
 520            elif isinstance(ir, Binary):
 521                if BinaryType.return_bool(ir.type):
 522                    ir.lvalue.set_type(ElementaryType("bool"))
 523                else:
 524                    ir.lvalue.set_type(ir.variable_left.type)
 525            elif isinstance(ir, Delete):
 526                # nothing to propagate
 527                pass
 528            elif isinstance(ir, LibraryCall):
 529                return convert_type_library_call(ir, ir.destination)
 530            elif isinstance(ir, HighLevelCall):
 531                t = ir.destination.type
 532                # Temporary operation (they are removed later)
 533                if t is None:
 534                    return None
 535
 536                if isinstance(t, ElementaryType) and t.name == "address":
 537                    if can_be_solidity_func(ir):
 538                        return convert_to_solidity_func(ir)
 539
 540                # convert library or top level function
 541                if t in using_for or "*" in using_for:
 542                    new_ir = convert_to_library_or_top_level(ir, node, using_for)
 543                    if new_ir:
 544                        return new_ir
 545
 546                if isinstance(t, UserDefinedType):
 547                    # UserdefinedType
 548                    t_type = t.type
 549                    if isinstance(t_type, Contract):
 550                        # the target contract of the IR is the t_type (the destination of the call)
 551                        return convert_type_of_high_and_internal_level_call(ir, t_type)
 552
 553                # Convert HighLevelCall to LowLevelCall
 554                if (isinstance(t, ElementaryType) and t.name == "address") or (
 555                    isinstance(t, TypeAlias) and t.underlying_type.name == "address"
 556                ):
 557                    # Cannot be a top level function with this.
 558                    assert isinstance(node_function, FunctionContract)
 559                    if ir.destination.name == "this":
 560                        # the target contract is the contract itself
 561                        return convert_type_of_high_and_internal_level_call(
 562                            ir, node_function.contract
 563                        )
 564                    if can_be_low_level(ir):
 565                        return convert_to_low_level(ir)
 566
 567                # Convert push operations
 568                # May need to insert a new operation
 569                # Which leads to return a list of operation
 570                if isinstance(t, ArrayType) or (
 571                    isinstance(t, ElementaryType) and t.type == "bytes"
 572                ):
 573                    if ir.function_name == "push" and len(ir.arguments) <= 1:
 574                        return convert_to_push(ir, node)
 575                    if ir.function_name == "pop" and len(ir.arguments) == 0:
 576                        return convert_to_pop(ir, node)
 577
 578            elif isinstance(ir, Index):
 579                if isinstance(ir.variable_left.type, MappingType):
 580                    ir.lvalue.set_type(ir.variable_left.type.type_to)
 581                elif isinstance(ir.variable_left.type, ArrayType):
 582                    ir.lvalue.set_type(ir.variable_left.type.type)
 583
 584            elif isinstance(ir, InitArray):
 585                length = len(ir.init_values)
 586                t = ir.init_values[0].type
 587                ir.lvalue.set_type(ArrayType(t, length))
 588            elif isinstance(ir, InternalCall):
 589                # if its not a tuple, return a singleton
 590                if ir.function is None:
 591                    func = node.function
 592                    function_contract = (
 593                        func.contract if isinstance(func, FunctionContract) else None
 594                    )
 595                    # the target contract might be None if its a top level function
 596                    convert_type_of_high_and_internal_level_call(ir, function_contract)
 597                return_type = ir.function.return_type
 598                if return_type:
 599                    if len(return_type) == 1:
 600                        ir.lvalue.set_type(return_type[0])
 601                    elif len(return_type) > 1:
 602                        ir.lvalue.set_type(return_type)
 603                else:
 604                    ir.lvalue = None
 605            elif isinstance(ir, InternalDynamicCall):
 606                # if its not a tuple, return a singleton
 607                return_type = ir.function_type.return_type
 608                if return_type:
 609                    if len(return_type) == 1:
 610                        ir.lvalue.set_type(return_type[0])
 611                    else:
 612                        ir.lvalue.set_type(return_type)
 613                else:
 614                    ir.lvalue = None
 615            elif isinstance(ir, LowLevelCall):
 616                # Call are not yet converted
 617                # This should not happen
 618                assert False
 619            elif isinstance(ir, Member):
 620                # TODO we should convert the reference to a temporary if the member is a length or a balance
 621                if (
 622                    ir.variable_right == "length"
 623                    and not isinstance(ir.variable_left, Contract)
 624                    and isinstance(ir.variable_left.type, (ElementaryType, ArrayType))
 625                ):
 626                    length = Length(ir.variable_left, ir.lvalue)
 627                    length.set_expression(ir.expression)
 628                    length.lvalue.points_to = ir.variable_left
 629                    length.set_node(ir.node)
 630                    return length
 631                # This only happen for .balance/code/codehash access on a variable for which we dont know at
 632                # early parsing time the type
 633                # Like
 634                # function return_addr() internal returns(addresss)
 635                #
 636                # return_addr().balance
 637                # Here slithIR will incorrectly create a REF variable instead of a Temp variable
 638                # However this pattern does not appear so often
 639                if (
 640                    ir.variable_right.name in ["balance", "code", "codehash"]
 641                    and not isinstance(ir.variable_left, Contract)
 642                    and isinstance(ir.variable_left.type, ElementaryType)
 643                ):
 644                    name = ir.variable_right.name + "(address)"
 645                    sol_func = SolidityFunction(name)
 646                    s = SolidityCall(
 647                        sol_func,
 648                        1,
 649                        ir.lvalue,
 650                        sol_func.return_type,
 651                    )
 652                    s.arguments.append(ir.variable_left)
 653                    s.set_expression(ir.expression)
 654                    s.lvalue.set_type(sol_func.return_type)
 655                    s.set_node(ir.node)

 662                    b = CodeSize(ir.variable_left, ir.lvalue)
 663                    b.set_expression(ir.expression)
 664                    b.set_node(ir.node)
 665                    return b
 666                if ir.variable_right == "selector" and isinstance(ir.variable_left, (CustomError)):
 667                    assignment = Assignment(
 668                        ir.lvalue,
 669                        Constant(

 675                    assignment.set_expression(ir.expression)
 676                    assignment.set_node(ir.node)

 681                    assignment = Assignment(
 682                        ir.lvalue,
 683                        Constant(
 684                            str(get_function_id(ir.variable_left.solidity_signature)),
 685                            ElementaryType("bytes4"),
 686                        ),
 687                        ElementaryType("bytes4"),

 689                    assignment.set_expression(ir.expression)
 690                    assignment.set_node(ir.node)

 693                if ir.variable_right == "selector" and isinstance(
 694                    ir.variable_left.type, (Function)
 695                ):
 696                    assignment = Assignment(
 697                        ir.lvalue,
 698                        Constant(str(get_function_id(ir.variable_left.type.full_name))),
 699                        ElementaryType("bytes4"),
 700                    )
 701                    assignment.set_expression(ir.expression)
 702                    assignment.set_node(ir.node)
 703                    assignment.lvalue.set_type(ElementaryType("bytes4"))
 704                    return assignment
 705                if isinstance(ir.variable_left, TemporaryVariable) and isinstance(
 706                    ir.variable_left.type, TypeInformation
 707                ):
 708                    return _convert_type_contract(ir)
 709                left = ir.variable_left
 710                t = None
 711                ir_func = ir.function
 712                # Handling of this.function_name usage
 713                if (
 714                    left == SolidityVariable("this")
 715                    and isinstance(ir.variable_right, Constant)
 716                    and isinstance(ir_func, FunctionContract)
 717                    and str(ir.variable_right) in [x.name for x in ir_func.contract.functions]
 718                ):
 719                    # Assumption that this.function_name can only compile if
 720                    # And the contract does not have two functions starting with function_name

 722                    # Error: Member "f" not unique after argument-dependent lookup in contract

 724                        (x for x in ir_func.contract.functions if x.name == str(ir.variable_right))

 728                elif isinstance(left, (Variable, SolidityVariable)):
 729                    t = ir.variable_left.type
 730                elif isinstance(left, (Contract, Enum, Structure)):
 731                    t = UserDefinedType(left)
 732                # can be None due to temporary operation
 733                if t:
 734                    if isinstance(t, UserDefinedType):
 735                        # UserdefinedType
 736                        type_t = t.type
 737                        if isinstance(type_t, Enum):
 738                            ir.lvalue.set_type(t)
 739                        elif isinstance(type_t, Structure):
 740                            elems = type_t.elems
 741                            for elem in elems:
 742                                if elem == ir.variable_right:
 743                                    ir.lvalue.set_type(elems[elem].type)
 744                        else:
 745                            assert isinstance(type_t, Contract)
 746                            # Allow type propagtion as a Function
 747                            # Only for reference variables
 748                            # This allows to track the selector keyword
 749                            # We dont need to check for function collision, as solc prevents the use of selector
 750                            # if there are multiple functions with the same name
 751                            f = next(
 752                                (f for f in type_t.functions if f.name == ir.variable_right),
 753                                None,
 754                            )
 755                            if f:
 756                                ir.lvalue.set_type(f)
 757                            else:
 758                                # Allow propgation for variable access through contract's name
 759                                # like Base_contract.my_variable

 761                                    (
 762                                        v
 763                                        for v in type_t.state_variables
 764                                        if v.name == ir.variable_right
 765                                    ),
 766                                    None,
 767                                )
 768                                if v:
 769                                    ir.lvalue.set_type(v.type)
 770            elif isinstance(ir, NewArray):
 771                ir.lvalue.set_type(ir.array_type)
 772            elif isinstance(ir, NewContract):
 773                contract = node.file_scope.get_contract_from_name(ir.contract_name)

 781            elif isinstance(ir, SolidityCall):
 782                if ir.function.name in ["type(address)", "type()"]:
 783                    ir.function.return_type = [TypeInformation(ir.arguments[0])]

 786                    ir.lvalue.set_type(return_type[0])

 801                    Argument,

 804                    TmpNewContract,

 810                pass

 816def extract_tmp_call(ins: TmpCall, contract: Optional[Contract]):  # pylint: disable=too-many-locals
 817    assert isinstance(ins, TmpCall)
 818    if isinstance(ins.called, Variable) and isinstance(ins.called.type, FunctionType):
 819        # If the call is made to a variable member, where the member is this
 820        # We need to convert it to a HighLelelCall and not an internal dynamic call
 821        if isinstance(ins.ori, Member) and ins.ori.variable_left == SolidityVariable("this"):
 822            pass
 823        else:
 824            call = InternalDynamicCall(ins.lvalue, ins.called, ins.called.type)
 825            call.set_expression(ins.expression)
 826            call.call_id = ins.call_id
 827            return call
 828    if isinstance(ins.ori, Member):
 829        # If there is a call on an inherited contract, it is an internal call or an event
 830        if contract and ins.ori.variable_left in contract.inheritance + [contract]:
 831            if str(ins.ori.variable_right) in [f.name for f in contract.functions]:
 832                internalcall = InternalCall(
 833                    (ins.ori.variable_right, ins.ori.variable_left.name),
 834                    ins.nbr_arguments,
 835                    ins.lvalue,
 836                    ins.type_call,
 837                )
 838                internalcall.set_expression(ins.expression)
 839                internalcall.call_id = ins.call_id
 840                return internalcall
 841            if str(ins.ori.variable_right) in [f.name for f in contract.events]:
 842                eventcall = EventCall(ins.ori.variable_right)
 843                eventcall.set_expression(ins.expression)
 844                eventcall.call_id = ins.call_id
 845                return eventcall
 846        if isinstance(ins.ori.variable_left, Contract):
 847            st = ins.ori.variable_left.get_structure_from_name(ins.ori.variable_right)
 848            if st:
 849                op = NewStructure(st, ins.lvalue)
 850                op.set_expression(ins.expression)
 851                op.call_id = ins.call_id
 852                return op
 853
 854            # For event emit through library
 855            # lib L { event E()}
 856            # ...
 857            # emit L.E();
 858            if str(ins.ori.variable_right) in [f.name for f in ins.ori.variable_left.events]:
 859                eventcall = EventCall(ins.ori.variable_right)
 860                eventcall.set_expression(ins.expression)
 861                eventcall.call_id = ins.call_id
 862                return eventcall
 863
 864            # lib Lib { error Error()} ... revert Lib.Error()
 865            if str(ins.ori.variable_right) in ins.ori.variable_left.custom_errors_as_dict:
 866                custom_error = ins.ori.variable_left.custom_errors_as_dict[
 867                    str(ins.ori.variable_right)
 868                ]
 869                assert isinstance(
 870                    custom_error,
 871                    CustomError,
 872                )
 873                sol_function = SolidityCustomRevert(custom_error)
 874                solidity_call = SolidityCall(
 875                    sol_function, ins.nbr_arguments, ins.lvalue, ins.type_call
 876                )
 877                solidity_call.set_expression(ins.expression)
 878                return solidity_call
 879
 880            libcall = LibraryCall(
 881                ins.ori.variable_left,
 882                ins.ori.variable_right,
 883                ins.nbr_arguments,
 884                ins.lvalue,
 885                ins.type_call,
 886            )
 887            libcall.set_expression(ins.expression)
 888            libcall.call_id = ins.call_id
 889            return libcall
 890        if isinstance(ins.ori.variable_left, Function):
 891            # Support for library call where the parameter is a function
 892            # We could merge this with the standard library handling
 893            # Except that we will have some troubles with using_for
 894            # As the type of the funciton will not match function()
 895            # Additionally we do not have a correct view on the parameters of the tmpcall
 896            # At this level
 897            #
 898            # library FunctionExtensions {
 899            #     function h(function() internal _t, uint8) internal {  }
 900            # }
 901            # contract FunctionMembers {
 902            #     using FunctionExtensions for function();
 903            #
 904            #     function f() public {
 905            #         f.h(1);
 906            #     }
 907            # }
 908            node_func = ins.node.function
 909            using_for = (
 910                node_func.contract.using_for_complete
 911                if isinstance(node_func, FunctionContract)
 912                else {}
 913            )
 914
 915            targeted_libraries = (
 916                [] + using_for.get("*", []) + using_for.get(FunctionType([], []), [])
 917            )
 918            lib_contract: Contract
 919            candidates = []
 920            for lib_contract_type in targeted_libraries:
 921                if not isinstance(lib_contract_type, UserDefinedType) and isinstance(
 922                    lib_contract_type.type, Contract
 923                ):
 924                    continue
 925                if isinstance(lib_contract_type, FunctionContract):
 926                    # Using for with list of functions, this is the function called
 927                    candidates.append(lib_contract_type)
 928                else:
 929                    lib_contract = lib_contract_type.type
 930                    for lib_func in lib_contract.functions:
 931                        if lib_func.name == ins.ori.variable_right:
 932                            candidates.append(lib_func)
 933
 934            if len(candidates) == 1:
 935                lib_func = candidates[0]
 936                # Library must be from a contract
 937                assert isinstance(lib_func, FunctionContract)
 938                lib_call = LibraryCall(
 939                    lib_func.contract,
 940                    Constant(lib_func.name),
 941                    len(lib_func.parameters),
 942                    ins.lvalue,
 943                    "d",
 944                )
 945                lib_call.set_expression(ins.expression)
 946                lib_call.set_node(ins.node)
 947                lib_call.call_gas = ins.call_gas
 948                lib_call.call_id = ins.call_id
 949                lib_call.set_node(ins.node)
 950                lib_call.function = lib_func
 951                lib_call.arguments.append(ins.ori.variable_left)
 952                return lib_call
 953            # We do not support something lik
 954            # library FunctionExtensions {
 955            #     function h(function() internal _t, uint8) internal {  }
 956            #     function h(function() internal _t, bool) internal {  }
 957            # }
 958            # contract FunctionMembers {
 959            #     using FunctionExtensions for function();
 960            #
 961            #     function f() public {
 962            #         f.h(1);
 963            #     }
 964            # }
 965            to_log = "Slither does not support dynamic functions to libraries if functions have the same name"
 966            to_log += f"{[candidate.full_name for candidate in candidates]}"
 967            raise SlithIRError(to_log)
 968        if isinstance(ins.ori.variable_left, SolidityImportPlaceHolder):
 969            # For top level import, where the import statement renames the filename
 970            # See https://blog.soliditylang.org/2020/09/02/solidity-0.7.1-release-announcement/
 971
 972            current_path = Path(ins.ori.variable_left.source_mapping.filename.absolute).parent
 973            target = str(
 974                Path(current_path, ins.ori.variable_left.import_directive.filename).absolute()
 975            )
 976            top_level_function_targets = [
 977                f
 978                for f in ins.compilation_unit.functions_top_level
 979                if f.source_mapping.filename.absolute == target
 980                and f.name == ins.ori.variable_right
 981                and len(f.parameters) == ins.nbr_arguments
 982            ]
 983
 984            internalcall = InternalCall(
 985                (ins.ori.variable_right, ins.ori.variable_left.name),
 986                ins.nbr_arguments,
 987                ins.lvalue,
 988                ins.type_call,
 989            )
 990            internalcall.set_expression(ins.expression)
 991            internalcall.call_id = ins.call_id
 992            internalcall.function_candidates = top_level_function_targets
 993            return internalcall
 994
 995        if ins.ori.variable_left == ElementaryType("bytes") and ins.ori.variable_right == Constant(
 996            "concat"
 997        ):
 998            s = SolidityCall(
 999                SolidityFunction("bytes.concat()"),
1000                ins.nbr_arguments,
1001                ins.lvalue,
1002                ins.type_call,
1003            )
1004            s.set_expression(ins.expression)
1005            return s
1006
1007        if ins.ori.variable_left == ElementaryType("string") and ins.ori.variable_right == Constant(
1008            "concat"
1009        ):
1010            s = SolidityCall(
1011                SolidityFunction("string.concat()"),
1012                ins.nbr_arguments,
1013                ins.lvalue,
1014                ins.type_call,
1015            )
1016            s.set_expression(ins.expression)
1017            return s
1018
1019        msgcall = HighLevelCall(
1020            ins.ori.variable_left,
1021            ins.ori.variable_right,
1022            ins.nbr_arguments,
1023            ins.lvalue,
1024            ins.type_call,
1025        )
1026        msgcall.call_id = ins.call_id
1027
1028        if ins.call_gas:
1029            msgcall.call_gas = ins.call_gas
1030        if ins.call_value:
1031            msgcall.call_value = ins.call_value
1032        msgcall.set_expression(ins.expression)
1033
1034        return msgcall
1035
1036    if isinstance(ins.ori, TmpCall):
1037        r = extract_tmp_call(ins.ori, contract)
1038        r.set_node(ins.node)
1039        return r
1040    if isinstance(ins.called, SolidityVariableComposed):
1041        if str(ins.called) == "block.blockhash":
1042            ins.called = SolidityFunction("blockhash(uint256)")
1043
1044    if isinstance(ins.called, SolidityFunction):
1045        s = SolidityCall(ins.called, ins.nbr_arguments, ins.lvalue, ins.type_call)
1046        s.set_expression(ins.expression)
1047        return s
1048
1049    if isinstance(ins.called, CustomError):
1050        sol_function = SolidityCustomRevert(ins.called)
1051        s = SolidityCall(sol_function, ins.nbr_arguments, ins.lvalue, ins.type_call)
1052        s.set_expression(ins.expression)
1053        return s
1054
1055    if isinstance(ins.ori, TmpNewElementaryType):
1056        n = NewElementaryType(ins.ori.type, ins.lvalue)
1057        n.set_expression(ins.expression)
1058        return n
1059
1060    if isinstance(ins.ori, TmpNewContract):
1061        op = NewContract(Constant(ins.ori.contract_name), ins.lvalue)
1062        op.set_expression(ins.expression)
1063        op.call_id = ins.call_id

1065            op.call_value = ins.call_value
1066        if ins.call_salt:
1067            op.call_salt = ins.call_salt
1068        return op
1069
1070    if isinstance(ins.ori, TmpNewArray):
1071        n = NewArray(ins.ori.depth, ins.ori.array_type, ins.lvalue)
1072        n.set_expression(ins.expression)
1073        return n
1074
1075    if isinstance(ins.called, Structure):
1076        op = NewStructure(ins.called, ins.lvalue)
1077        op.set_expression(ins.expression)
1078        op.call_id = ins.call_id
1079        op.set_expression(ins.expression)
1080        return op
1081
1082    if isinstance(ins.called, Event):
1083        e = EventCall(ins.called.name)
1084        e.set_expression(ins.expression)
1085        return e
1086
1087    if isinstance(ins.called, Contract):
1088        # Called a base constructor, where there is no constructor
1089        if ins.called.constructor is None:
1090            return Nop()
1091        # Case where:
1092        # contract A{ constructor(uint) }
1093        # contract B is A {}
1094        # contract C is B{ constructor() A(10) B() {}
1095        # C calls B(), which does not exist
1096        # Ideally we should compare here for the parameters types too

1100            ins.called.constructor, ins.nbr_arguments, ins.lvalue, ins.type_call
1101        )
1102        internalcall.call_id = ins.call_id

1104        return internalcall
1105
1106    raise Exception(f"Not extracted {type(ins.called)} {ins}")
