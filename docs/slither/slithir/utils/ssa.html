 71def transform_slithir_vars_to_ssa(function):
 72    """
 73    Transform slithIR vars to SSA (TemporaryVariable, ReferenceVariable, TupleVariable)
 74    """
 75    variables = []
 76    for node in function.nodes:
 77        for ir in node.irs_ssa:
 78            if isinstance(ir, OperationWithLValue) and not ir.lvalue in variables:
 79                variables += [ir.lvalue]
 80
 81    tmp_variables = [v for v in variables if isinstance(v, TemporaryVariable)]
 82    for idx, _ in enumerate(tmp_variables):
 83        tmp_variables[idx].index = idx
 84    ref_variables = [v for v in variables if isinstance(v, ReferenceVariable)]
 85    for idx, _ in enumerate(ref_variables):
 86        ref_variables[idx].index = idx
 87    tuple_variables = [v for v in variables if isinstance(v, TupleVariable)]
 88    for idx, _ in enumerate(tuple_variables):
 89        tuple_variables[idx].index = idx

101def add_ssa_ir(function, all_state_variables_instances):
102    """
103        Add SSA version of the IR
104    Args:
105        function
106        all_state_variables_instances
107    """
108
109    if not function.is_implemented:
110        return
111
112    init_definition = {}
113    for v in function.parameters:
114        if v.name:
115            init_definition[v.name] = (v, function.entry_point)
116            function.entry_point.add_ssa_ir(Phi(LocalIRVariable(v), set()))
117
118    for v in function.returns:
119        if v.name:
120            init_definition[v.name] = (v, function.entry_point)
121
122    # We only add phi function for state variable at entry node if
123    # The state variable is used
124    # And if the state variables is written in another function (otherwise its stay at index 0)
125    for (_, variable_instance) in all_state_variables_instances.items():
126        if is_used_later(function.entry_point, variable_instance):
127            # rvalues are fixed in solc_parsing.declaration.function
128            function.entry_point.add_ssa_ir(Phi(StateIRVariable(variable_instance), set()))
129
130    add_phi_origins(function.entry_point, init_definition, {})
131
132    for node in function.nodes:
133        for (variable, nodes) in node.phi_origins_local_variables.values():
134            if len(nodes) < 2:
135                continue
136            if not is_used_later(node, variable):
137                continue
138            node.add_ssa_ir(Phi(LocalIRVariable(variable), nodes))
139        for (variable, nodes) in node.phi_origins_state_variables.values():
140            if len(nodes) < 2:
141                continue
142            # if not is_used_later(node, variable.name, []):
143            #    continue
144            node.add_ssa_ir(Phi(StateIRVariable(variable), nodes))
145
146    init_local_variables_instances = {}
147    for v in function.parameters:
148        if v.name:
149            new_var = LocalIRVariable(v)
150            function.add_parameter_ssa(new_var)
151            if new_var.is_storage:
152                fake_variable = LocalIRVariable(v)
153                fake_variable.name = "STORAGE_" + fake_variable.name
154                fake_variable.set_location("reference_to_storage")
155                new_var.refers_to = {fake_variable}
156                init_local_variables_instances[fake_variable.name] = fake_variable
157            init_local_variables_instances[v.name] = new_var
158
159    for v in function.returns:
160        if v.name:
161            new_var = LocalIRVariable(v)
162            function.add_return_ssa(new_var)
163            if new_var.is_storage:
164                fake_variable = LocalIRVariable(v)
165                fake_variable.name = "STORAGE_" + fake_variable.name
166                fake_variable.set_location("reference_to_storage")
167                new_var.refers_to = {fake_variable}
168                init_local_variables_instances[fake_variable.name] = fake_variable
169            init_local_variables_instances[v.name] = new_var
170
171    all_init_local_variables_instances = dict(init_local_variables_instances)
172
173    init_state_variables_instances = dict(all_state_variables_instances)
174
175    initiate_all_local_variables_instances(
176        function.nodes,
177        init_local_variables_instances,
178        all_init_local_variables_instances,
179    )
180
181    generate_ssa_irs(
182        function.entry_point,
183        dict(init_local_variables_instances),
184        all_init_local_variables_instances,
185        dict(init_state_variables_instances),
186        all_state_variables_instances,
187        init_local_variables_instances,
188        [],
189    )
190
191    fix_phi_rvalues_and_storage_ref(
192        function.entry_point,
193        dict(init_local_variables_instances),
194        all_init_local_variables_instances,
195        dict(init_state_variables_instances),
196        all_state_variables_instances,
197        init_local_variables_instances,
198    )

201def generate_ssa_irs(
202    node,
203    local_variables_instances,
204    all_local_variables_instances,
205    state_variables_instances,
206    all_state_variables_instances,
207    init_local_variables_instances,
208    visited,
209):
210
211    if node in visited:
212        return
213
214    if node.type in [NodeType.ENDIF, NodeType.ENDLOOP] and any(
215        not father in visited for father in node.fathers
216    ):
217        return
218
219    # visited is shared
220    visited.append(node)
221
222    for ir in node.irs_ssa:
223        assert isinstance(ir, Phi)
224        update_lvalue(
225            ir,
226            node,
227            local_variables_instances,
228            all_local_variables_instances,
229            state_variables_instances,
230            all_state_variables_instances,
231        )
232
233    # these variables are lived only during the liveness of the block
234    # They dont need phi function
235    temporary_variables_instances = {}
236    reference_variables_instances = {}
237    tuple_variables_instances = {}
238
239    for ir in node.irs:
240        new_ir = copy_ir(
241            ir,
242            local_variables_instances,
243            state_variables_instances,
244            temporary_variables_instances,
245            reference_variables_instances,
246            tuple_variables_instances,
247            all_local_variables_instances,
248        )
249
250        new_ir.set_expression(ir.expression)
251        new_ir.set_node(ir.node)
252
253        update_lvalue(
254            new_ir,
255            node,
256            local_variables_instances,
257            all_local_variables_instances,
258            state_variables_instances,
259            all_state_variables_instances,
260        )
261
262        if new_ir:
263
264            node.add_ssa_ir(new_ir)
265
266            if isinstance(ir, (InternalCall, HighLevelCall, InternalDynamicCall, LowLevelCall)):
267                if isinstance(ir, LibraryCall):
268                    continue
269                for variable in all_state_variables_instances.values():
270                    if not is_used_later(node, variable):
271                        continue
272                    new_var = StateIRVariable(variable)
273                    new_var.index = all_state_variables_instances[variable.canonical_name].index + 1
274                    all_state_variables_instances[variable.canonical_name] = new_var
275                    state_variables_instances[variable.canonical_name] = new_var
276                    phi_ir = PhiCallback(new_var, {node}, new_ir, variable)
277                    # rvalues are fixed in solc_parsing.declaration.function
278                    node.add_ssa_ir(phi_ir)
279
280            if isinstance(new_ir, (Assignment, Binary)):
281                if isinstance(new_ir.lvalue, LocalIRVariable):
282                    if new_ir.lvalue.is_storage:
283                        if isinstance(new_ir.rvalue, ReferenceVariable):
284                            refers_to = new_ir.rvalue.points_to_origin
285                            new_ir.lvalue.add_refers_to(refers_to)
286                        # Discard Constant
287                        # This can happen on yul, like
288                        # assembly { var.slot = some_value }
289                        # Here we do not keep track of the references as we do not track
290                        # such low level manipulation
291                        # However we could extend our storage model to do so in the future
292                        elif not isinstance(new_ir.rvalue, Constant):
293                            new_ir.lvalue.add_refers_to(new_ir.rvalue)
294
295    for succ in node.dominator_successors:
296        generate_ssa_irs(
297            succ,
298            dict(local_variables_instances),
299            all_local_variables_instances,
300            dict(state_variables_instances),
301            all_state_variables_instances,
302            init_local_variables_instances,
303            visited,
304        )
305
306    for dominated in node.dominance_frontier:
307        generate_ssa_irs(
308            dominated,
309            dict(local_variables_instances),
310            all_local_variables_instances,
311            dict(state_variables_instances),
312            all_state_variables_instances,
313            init_local_variables_instances,
314            visited,
315        )

326def last_name(n, var, init_vars):
327    candidates = []
328    # Todo optimize by creating a variables_ssa_written attribute
329    for ir_ssa in n.irs_ssa:
330        if isinstance(ir_ssa, OperationWithLValue):
331            lvalue = ir_ssa.lvalue
332            while isinstance(lvalue, ReferenceVariable):
333                lvalue = lvalue.points_to
334            if lvalue and lvalue.name == var.name:
335                candidates.append(lvalue)
336    if n.variable_declaration and n.variable_declaration.name == var.name:
337        candidates.append(LocalIRVariable(n.variable_declaration))
338    if n.type == NodeType.ENTRYPOINT:
339        if var.name in init_vars:
340            candidates.append(init_vars[var.name])
341    assert candidates
342    return max(candidates, key=lambda v: v.index)

345def is_used_later(initial_node, variable):
346    # TODO: does not handle the case where its read and written in the declaration node
347    # It can be problematic if this happens in a loop/if structure
348    # Ex:
349    # for(;true;){
350    #   if(true){
351    #     uint a = a;
352    #    }
353    #     ..
354    to_explore = {initial_node}
355    explored = set()
356
357    while to_explore:
358        node = to_explore.pop()
359        explored.add(node)
360        if isinstance(variable, LocalVariable):
361            if any(v.name == variable.name for v in node.local_variables_read):
362                return True
363            if any(v.name == variable.name for v in node.local_variables_written):
364                return False
365        if isinstance(variable, StateVariable):
366            if any(
367                v.name == variable.name and v.contract == variable.contract
368                for v in node.state_variables_read
369            ):
370                return True
371            if any(
372                v.name == variable.name and v.contract == variable.contract
373                for v in node.state_variables_written
374            ):
375                return False
376        for son in node.sons:
377            if not son in explored:
378                to_explore.add(son)
379
380    return False

391def update_lvalue(
392    new_ir,
393    node,
394    local_variables_instances,
395    all_local_variables_instances,
396    state_variables_instances,
397    all_state_variables_instances,
398):
399    if isinstance(new_ir, OperationWithLValue):
400        lvalue = new_ir.lvalue
401        update_through_ref = False
402        if isinstance(new_ir, (Assignment, Binary)):
403            if isinstance(lvalue, ReferenceVariable):
404                update_through_ref = True
405                while isinstance(lvalue, ReferenceVariable):
406                    lvalue = lvalue.points_to
407        if isinstance(lvalue, (LocalIRVariable, StateIRVariable)):
408            if isinstance(lvalue, LocalIRVariable):
409                new_var = LocalIRVariable(lvalue)
410                new_var.index = all_local_variables_instances[lvalue.name].index + 1
411                all_local_variables_instances[lvalue.name] = new_var
412                local_variables_instances[lvalue.name] = new_var
413            else:
414                new_var = StateIRVariable(lvalue)
415                new_var.index = all_state_variables_instances[lvalue.canonical_name].index + 1
416                all_state_variables_instances[lvalue.canonical_name] = new_var
417                state_variables_instances[lvalue.canonical_name] = new_var
418            if update_through_ref:
419                phi_operation = Phi(new_var, {node})
420                phi_operation.rvalues = [lvalue]
421                node.add_ssa_ir(phi_operation)
422            if not isinstance(new_ir.lvalue, ReferenceVariable):
423                new_ir.lvalue = new_var
424            else:
425                to_update = new_ir.lvalue
426                while isinstance(to_update.points_to, ReferenceVariable):
427                    to_update = to_update.points_to
428                to_update.points_to = new_var

459def fix_phi_rvalues_and_storage_ref(
460    node,
461    local_variables_instances,
462    all_local_variables_instances,
463    state_variables_instances,
464    all_state_variables_instances,
465    init_local_variables_instances,
466):
467    for ir in node.irs_ssa:
468        if isinstance(ir, (Phi)) and not ir.rvalues:
469            variables = [
470                last_name(dst, ir.lvalue, init_local_variables_instances) for dst in ir.nodes
471            ]
472            ir.rvalues = variables
473        if isinstance(ir, (Phi, PhiCallback)):
474            if isinstance(ir.lvalue, LocalIRVariable):
475                if ir.lvalue.is_storage:
476                    l = [v.refers_to for v in ir.rvalues]
477                    l = [item for sublist in l for item in sublist]
478                    ir.lvalue.refers_to = set(l)
479
480        if isinstance(ir, (Assignment, Binary)):
481            if isinstance(ir.lvalue, ReferenceVariable):
482                origin = ir.lvalue.points_to_origin
483
484                if isinstance(origin, LocalIRVariable):
485                    if origin.is_storage:
486                        for refers_to in origin.refers_to:
487                            phi_ir = Phi(refers_to, {node})
488                            phi_ir.rvalues = [origin]
489                            node.add_ssa_ir(phi_ir)
490                            update_lvalue(
491                                phi_ir,
492                                node,
493                                local_variables_instances,
494                                all_local_variables_instances,
495                                state_variables_instances,
496                                all_state_variables_instances,
497                            )
498    for succ in node.dominator_successors:
499        fix_phi_rvalues_and_storage_ref(
500            succ,
501            dict(local_variables_instances),
502            all_local_variables_instances,
503            dict(state_variables_instances),
504            all_state_variables_instances,
505            init_local_variables_instances,
506        )

509def add_phi_origins(node, local_variables_definition, state_variables_definition):
510
511    # Add new key to local_variables_definition
512    # The key is the variable_name
513    # The value is (variable_instance, the node where its written)
514    # We keep the instance as we want to avoid to add __hash__ on v.name in Variable
515    # That might work for this used, but could create collision for other uses
516    local_variables_definition = dict(
517        local_variables_definition,
518        **{v.name: (v, node) for v in node.local_variables_written},
519    )
520    state_variables_definition = dict(
521        state_variables_definition,
522        **{v.canonical_name: (v, node) for v in node.state_variables_written},
523    )
524
525    # For unini variable declaration
526    if (
527        node.variable_declaration
528        and not node.variable_declaration.name in local_variables_definition
529    ):
530        local_variables_definition[node.variable_declaration.name] = (
531            node.variable_declaration,
532            node,
533        )
534
535    # filter length of successors because we have node with one successor
536    # while most of the ssa textbook would represent following nodes as one
537    if node.dominance_frontier and len(node.dominator_successors) != 1:
538        for phi_node in node.dominance_frontier:
539            for _, (variable, n) in local_variables_definition.items():
540                phi_node.add_phi_origin_local_variable(variable, n)
541            for _, (variable, n) in state_variables_definition.items():
542                phi_node.add_phi_origin_state_variable(variable, n)
543
544    if not node.dominator_successors:
545        return
546    for succ in node.dominator_successors:
547        add_phi_origins(succ, local_variables_definition, state_variables_definition)

558def get(
559    variable,
560    local_variables_instances,
561    state_variables_instances,
562    temporary_variables_instances,
563    reference_variables_instances,
564    tuple_variables_instances,
565    all_local_variables_instances,
566):
567    # variable can be None
568    # for example, on LowLevelCall, ir.lvalue can be none
569    if variable is None:
570        return None
571    if isinstance(variable, LocalVariable):
572        if variable.name in local_variables_instances:
573            return local_variables_instances[variable.name]
574        new_var = LocalIRVariable(variable)
575        local_variables_instances[variable.name] = new_var
576        all_local_variables_instances[variable.name] = new_var
577        return new_var
578    if isinstance(variable, StateVariable) and variable.canonical_name in state_variables_instances:
579        return state_variables_instances[variable.canonical_name]
580    if isinstance(variable, ReferenceVariable):
581        if not variable.index in reference_variables_instances:
582            new_variable = ReferenceVariableSSA(variable)
583            if variable.points_to:
584                new_variable.points_to = get(
585                    variable.points_to,
586                    local_variables_instances,
587                    state_variables_instances,
588                    temporary_variables_instances,
589                    reference_variables_instances,
590                    tuple_variables_instances,
591                    all_local_variables_instances,
592                )
593            new_variable.set_type(variable.type)
594            reference_variables_instances[variable.index] = new_variable
595        return reference_variables_instances[variable.index]
596    if isinstance(variable, TemporaryVariable):
597        if not variable.index in temporary_variables_instances:
598            new_variable = TemporaryVariableSSA(variable)
599            new_variable.set_type(variable.type)
600            temporary_variables_instances[variable.index] = new_variable
601        return temporary_variables_instances[variable.index]
602    if isinstance(variable, TupleVariable):
603        if not variable.index in tuple_variables_instances:
604            new_variable = TupleVariableSSA(variable)
605            new_variable.set_type(variable.type)
606            tuple_variables_instances[variable.index] = new_variable
607        return tuple_variables_instances[variable.index]
608    assert isinstance(
609        variable,
610        (
611            Constant,
612            SolidityVariable,
613            Contract,
614            Enum,
615            SolidityFunction,
616            Structure,
617            Function,
618            Type,
619            SolidityImportPlaceHolder,
620            TopLevelVariable,
621        ),
622    )  # type for abi.decode(.., t)
623    return variable

633def _get_traversal(values, *instances):
634    ret = []

657def copy_ir(ir, *instances):
658    """
659    Args:
660        ir (Operation)
661        local_variables_instances(dict(str -> LocalVariable))
662        state_variables_instances(dict(str -> StateVariable))
663        temporary_variables_instances(dict(int -> Variable))
664        reference_variables_instances(dict(int -> Variable))
665
666    Note: temporary and reference can be indexed by int, as they dont need phi functions
667    """
668    if isinstance(ir, Assignment):
669        lvalue = get_variable(ir, lambda x: x.lvalue, *instances)
670        rvalue = get_variable(ir, lambda x: x.rvalue, *instances)
671        variable_return_type = ir.variable_return_type
672        return Assignment(lvalue, rvalue, variable_return_type)
673    if isinstance(ir, Binary):
674        lvalue = get_variable(ir, lambda x: x.lvalue, *instances)
675        variable_left = get_variable(ir, lambda x: x.variable_left, *instances)
676        variable_right = get_variable(ir, lambda x: x.variable_right, *instances)
677        operation_type = ir.type
678        return Binary(lvalue, variable_left, variable_right, operation_type)
679    if isinstance(ir, CodeSize):
680        lvalue = get_variable(ir, lambda x: x.lvalue, *instances)
681        value = get_variable(ir, lambda x: x.value, *instances)
682        return CodeSize(value, lvalue)
683    if isinstance(ir, Condition):
684        val = get_variable(ir, lambda x: x.value, *instances)
685        return Condition(val)
686    if isinstance(ir, Delete):
