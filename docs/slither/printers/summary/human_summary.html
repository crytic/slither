168    def _lines_number(self):
169        if not self.slither.source_code:
170            return None
171        total_dep_lines = 0
172        total_lines = 0
173        total_tests_lines = 0
174
175        for filename, source_code in self.slither.source_code.items():
176            lines = len(source_code.splitlines())
177            is_dep = False
178            if self.slither.crytic_compile:
179                is_dep = self.slither.crytic_compile.is_dependency(filename)
180            if is_dep:
181                total_dep_lines += lines
182            else:
183                if is_test_file(Path(filename)):
184                    total_tests_lines += lines
185                else:
186                    total_lines += lines
187        return total_lines, total_dep_lines, total_tests_lines

189    def _get_number_of_assembly_lines(self):
190        total_asm_lines = 0
191        for contract in self.contracts:
192            for function in contract.functions_declared:
193                for node in function.nodes:
194                    if node.type == NodeType.ASSEMBLY:
195                        inline_asm = node.inline_asm
196                        if inline_asm:
197                            total_asm_lines += len(inline_asm.splitlines())
198        return total_asm_lines

294    def output(self, _filename):  # pylint: disable=too-many-locals,too-many-statements
295        """
296        _filename is not used
297            Args:
298                _filename(string)
299        """
300
301        txt = "\n"
302        txt += self._compilation_type()
303
304        results = {
305            "contracts": {"elements": []},
306            "number_lines": 0,
307            "number_lines_in_dependencies": 0,
308            "number_lines_assembly": 0,
309            "standard_libraries": [],
310            "ercs": [],
311            "number_findings": {},
312            "detectors": [],
313        }
314
315        lines_number = self._lines_number()
316        if lines_number:
317            total_lines, total_dep_lines, total_tests_lines = lines_number
318            txt += f"Number of lines: {total_lines} (+ {total_dep_lines} in dependencies, + {total_tests_lines} in tests)\n"
319            results["number_lines"] = total_lines
320            results["number_lines__dependencies"] = total_dep_lines
321            total_asm_lines = self._get_number_of_assembly_lines()
322            txt += f"Number of assembly lines: {total_asm_lines}\n"
323            results["number_lines_assembly"] = total_asm_lines
324
325        (
326            number_contracts,
327            number_contracts_deps,
328            number_contracts_tests,
329        ) = self._number_contracts()
330        txt += f"Number of contracts: {number_contracts} (+ {number_contracts_deps} in dependencies, + {number_contracts_tests} tests) \n\n"
331
332        (
333            txt_detectors,
334            detectors_results,
335            optimization,
336            info,
337            low,
338            medium,
339            high,
340        ) = self.get_detectors_result()
341        txt += txt_detectors
342
343        results["number_findings"] = {
344            "optimization_issues": optimization,
345            "informational_issues": info,
346            "low_issues": low,
347            "medium_issues": medium,
348            "high_issues": high,
349        }
350        results["detectors"] = detectors_results
351
352        libs = self._standard_libraries()
353        if libs:
354            txt += f'\nUse: {", ".join(libs)}\n'
355            results["standard_libraries"] = [str(l) for l in libs]
356
357        ercs = self._ercs()
358        if ercs:
359            txt += f'ERCs: {", ".join(ercs)}\n'
360            results["ercs"] = [str(e) for e in ercs]
361
362        table = MyPrettyTable(
363            ["Name", "# functions", "ERCS", "ERC20 info", "Complex code", "Features"]
364        )
365        for contract in self.slither.contracts_derived:
366
367            if contract.is_from_dependency() or contract.is_test:
368                continue
369
370            is_complex = self.is_complex_code(contract)
371            number_functions = self._number_functions(contract)
372            ercs = ",".join(contract.ercs())
373            is_erc20 = contract.is_erc20()
374            erc20_info = ""
375            if is_erc20:
376                erc20_info += self.get_summary_erc20(contract)
377
378            features = "\n".join(
379                [name for name, to_print in self._get_features(contract).items() if to_print]
380            )
381
382            table.add_row(
383                [
384                    contract.name,
385                    number_functions,
386                    ercs,
387                    erc20_info,
388                    is_complex,
389                    features,
390                ]
391            )
392
393        self.info(txt + "\n" + str(table))
394
395        results_contract = output.Output("")
396        for contract in self.slither.contracts_derived:
397            if contract.is_test or contract.is_from_dependency():
398                continue
399
400            contract_d = {
401                "contract_name": contract.name,
402                "is_complex_code": self._is_complex_code(contract),
403                "is_erc20": contract.is_erc20(),
404                "number_functions": self._number_functions(contract),
405                "features": [
406                    name for name, to_print in self._get_features(contract).items() if to_print
407                ],
408            }
409            if contract_d["is_erc20"]:
410                pause, mint_limited, race_condition_mitigated = self._get_summary_erc20(contract)
411                contract_d["erc20_pause"] = pause
412                if mint_limited is not None:
413                    contract_d["erc20_can_mint"] = True
414                    contract_d["erc20_mint_limited"] = mint_limited
415                else:
416                    contract_d["erc20_can_mint"] = False
417                contract_d["erc20_race_condition_mitigated"] = race_condition_mitigated
418            results_contract.add_contract(contract, additional_fields=contract_d)
419
420        results["contracts"]["elements"] = results_contract.elements
421
422        json = self.generate_output(txt, additional_fields=results)
423
424        return json
