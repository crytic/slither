 144    def _add_local_variable(
 145        self, local_var_parser: Union[LocalVariableSolc, LocalVariableInitFromTupleSolc]
 146    ):
 147        # If two local variables have the same name
 148        # We add a suffix to the new variable
 149        # This is done to prevent collision during SSA translation
 150        # Use of while in case of collision
 151        # In the worst case, the name will be really long
 152        if local_var_parser.underlying_variable.name:
 153            known_variables = [v.name for v in self._function.variables]
 154            while local_var_parser.underlying_variable.name in known_variables:
 155                local_var_parser.underlying_variable.name += (
 156                    f"_scope_{self._counter_scope_local_variables}"
 157                )
 158                self._counter_scope_local_variables += 1
 159                known_variables = [v.name for v in self._function.variables]
 160        if local_var_parser.reference_id is not None:
 161            self._variables_renamed[local_var_parser.reference_id] = local_var_parser
 162        self._function.variables_as_dict[
 163            local_var_parser.underlying_variable.name
 164        ] = local_var_parser.underlying_variable
 165        self._local_variables_parser.append(local_var_parser)

 178    def _analyze_type(self):
 179        """
 180        Analyz the type of the function
 181        Myst be called in the constructor as the name might change according to the function's type
 182        For example both the fallback and the receiver will have an empty name
 183        :return:
 184        """
 185        if self.is_compact_ast:
 186            attributes = self._functionNotParsed
 187        else:
 188            attributes = self._functionNotParsed["attributes"]
 189
 190        if self._function.name == "":
 191            self._function.function_type = FunctionType.FALLBACK
 192            # 0.6.x introduced the receiver function
 193            # It has also an empty name, so we need to check the kind attribute
 194            if "kind" in attributes:
 195                if attributes["kind"] == "receive":
 196                    self._function.function_type = FunctionType.RECEIVE
 197        else:
 198            self._function.function_type = FunctionType.NORMAL
 199
 200        if isinstance(self._function, FunctionContract):
 201            if self._function.name == self._function.contract_declarer.name:
 202                self._function.function_type = FunctionType.CONSTRUCTOR

 204    def _analyze_attributes(self):
 205        if self.is_compact_ast:
 206            attributes = self._functionNotParsed
 207        else:
 208            attributes = self._functionNotParsed["attributes"]
 209
 210        if "payable" in attributes:
 211            self._function.payable = attributes["payable"]
 212        if "stateMutability" in attributes:
 213            if attributes["stateMutability"] == "payable":
 214                self._function.payable = True
 215            elif attributes["stateMutability"] == "pure":
 216                self._function.pure = True
 217                self._function.view = True
 218            elif attributes["stateMutability"] == "view":
 219                self._function.view = True
 220
 221        if "constant" in attributes:
 222            self._function.view = attributes["constant"]
 223
 224        if "isConstructor" in attributes and attributes["isConstructor"]:
 225            self._function.function_type = FunctionType.CONSTRUCTOR
 226
 227        if "kind" in attributes:
 228            if attributes["kind"] == "constructor":
 229                self._function.function_type = FunctionType.CONSTRUCTOR
 230
 231        if "visibility" in attributes:
 232            self._function.visibility = attributes["visibility"]
 233        # old solc
 234        elif "public" in attributes:
 235            if attributes["public"]:
 236                self._function.visibility = "public"
 237            else:
 238                self._function.visibility = "private"
 239        else:
 240            self._function.visibility = "public"
 241
 242        if "payable" in attributes:
 243            self._function.payable = attributes["payable"]

 245    def analyze_params(self):
 246        # Can be re-analyzed due to inheritance
 247        if self._params_was_analyzed:
 248            return
 249
 250        self._params_was_analyzed = True
 251
 252        self._analyze_attributes()
 253
 254        if self.is_compact_ast:
 255            params = self._functionNotParsed["parameters"]
 256            returns = self._functionNotParsed["returnParameters"]
 257        else:
 258            children = self._functionNotParsed[self.get_children("children")]
 259            # It uses to be
 260            # params = children[0]
 261            # returns = children[1]
 262            # But from Solidity 0.6.3 to 0.6.10 (included)
 263            # Comment above a function might be added in the children
 264            child_iter = iter(
 265                [child for child in children if child[self.get_key()] == "ParameterList"]
 266            )
 267            params = next(child_iter)
 268            returns = next(child_iter)
 269
 270        if params:
 271            self._parse_params(params)
 272        if returns:
 273            self._parse_returns(returns)
 274
 275    def analyze_content(self):
 276        if self._content_was_analyzed:
 277            return
 278
 279        self._content_was_analyzed = True
 280
 281        if self.is_compact_ast:
 282            body = self._functionNotParsed.get("body", None)
 283
 284            if body and body[self.get_key()] == "Block":
 285                self._function.is_implemented = True
 286                self._parse_cfg(body)
 287
 288            for modifier in self._functionNotParsed["modifiers"]:
 289                self._parse_modifier(modifier)
 290
 291        else:
 292            children = self._functionNotParsed[self.get_children("children")]
 293            self._function.is_implemented = False
 294            for child in children[2:]:
 295                if child[self.get_key()] == "Block":
 296                    self._function.is_implemented = True
 297                    self._parse_cfg(child)
 298
 299            # Parse modifier after parsing all the block
 300            # In the case a local variable is used in the modifier
 301            for child in children[2:]:
 302                if child[self.get_key()] == "ModifierInvocation":
 303                    self._parse_modifier(child)
 304
 305        for local_var_parser in self._local_variables_parser:
 306            local_var_parser.analyze(self)
 307
 308        for node_parser in self._node_to_nodesolc.values():
 309            node_parser.analyze_expressions(self)
 310
 311        for node_parser in self._node_to_yulobject.values():
 312            node_parser.analyze_expressions()
 313
 314        self._rewrite_ternary_as_if_else()
 315
 316        self._remove_alone_endif()

 357    def _parse_if(self, if_statement: Dict, node: NodeSolc) -> NodeSolc:
 358        # IfStatement = 'if' '(' Expression ')' Statement ( 'else' Statement )?
 359        falseStatement = None
 360
 361        if self.is_compact_ast:
 362            condition = if_statement["condition"]
 363            # Note: check if the expression could be directly
 364            # parsed here
 365            condition_node = self._new_node(
 366                NodeType.IF, condition["src"], node.underlying_node.scope
 367            )
 368            condition_node.add_unparsed_expression(condition)
 369            link_underlying_nodes(node, condition_node)
 370            true_scope = Scope(
 371                node.underlying_node.scope.is_checked, False, node.underlying_node.scope
 372            )
 373            trueStatement = self._parse_statement(
 374                if_statement["trueBody"], condition_node, true_scope
 375            )
 376            if "falseBody" in if_statement and if_statement["falseBody"]:
 377                false_scope = Scope(
 378                    node.underlying_node.scope.is_checked, False, node.underlying_node.scope
 379                )
 380                falseStatement = self._parse_statement(
 381                    if_statement["falseBody"], condition_node, false_scope
 382                )
 383        else:
 384            children = if_statement[self.get_children("children")]
 385            condition = children[0]
 386            # Note: check if the expression could be directly
 387            # parsed here
 388            condition_node = self._new_node(
 389                NodeType.IF, condition["src"], node.underlying_node.scope
 390            )
 391            condition_node.add_unparsed_expression(condition)
 392            link_underlying_nodes(node, condition_node)
 393            true_scope = Scope(
 394                node.underlying_node.scope.is_checked, False, node.underlying_node.scope
 395            )
 396            trueStatement = self._parse_statement(children[1], condition_node, true_scope)
 397            if len(children) == 3:
 398                false_scope = Scope(
 399                    node.underlying_node.scope.is_checked, False, node.underlying_node.scope
 400                )
 401                falseStatement = self._parse_statement(children[2], condition_node, false_scope)
 402
 403        endIf_node = self._new_node(NodeType.ENDIF, if_statement["src"], node.underlying_node.scope)
 404        link_underlying_nodes(trueStatement, endIf_node)
 405
 406        if falseStatement:
 407            link_underlying_nodes(falseStatement, endIf_node)
 408        else:
 409            link_underlying_nodes(condition_node, endIf_node)
 410        return endIf_node

 412    def _parse_while(self, whilte_statement: Dict, node: NodeSolc) -> NodeSolc:
 413        # WhileStatement = 'while' '(' Expression ')' Statement
 414
 415        node_startWhile = self._new_node(
 416            NodeType.STARTLOOP, whilte_statement["src"], node.underlying_node.scope
 417        )
 418
 419        body_scope = Scope(node.underlying_node.scope.is_checked, False, node.underlying_node.scope)
 420        if self.is_compact_ast:
 421            node_condition = self._new_node(
 422                NodeType.IFLOOP, whilte_statement["condition"]["src"], node.underlying_node.scope
 423            )
 424            node_condition.add_unparsed_expression(whilte_statement["condition"])
 425            statement = self._parse_statement(whilte_statement["body"], node_condition, body_scope)
 426        else:
 427            children = whilte_statement[self.get_children("children")]
 428            expression = children[0]
 429            node_condition = self._new_node(
 430                NodeType.IFLOOP, expression["src"], node.underlying_node.scope
 431            )
 432            node_condition.add_unparsed_expression(expression)
 433            statement = self._parse_statement(children[1], node_condition, body_scope)
 434
 435        node_endWhile = self._new_node(
 436            NodeType.ENDLOOP, whilte_statement["src"], node.underlying_node.scope
 437        )
 438
 439        link_underlying_nodes(node, node_startWhile)
 440        link_underlying_nodes(node_startWhile, node_condition)
 441        link_underlying_nodes(statement, node_condition)
 442        link_underlying_nodes(node_condition, node_endWhile)
 443
 444        return node_endWhile

 456    def _parse_for_legacy_ast(
 457        self, statement: Dict
 458    ) -> (Optional[Dict], Optional[Dict], Optional[Dict], Dict):
 459        # if we're using an old version of solc (anything below and including 0.4.11) or if the user
 460        # explicitly enabled compact ast, we might need to make some best-effort guesses
 461        children = statement[self.get_children("children")]
 462
 463        # there should always be at least one, and never more than 4, children
 464        assert 1 <= len(children) <= 4
 465
 466        # the last element of the children array must be the body, since it's mandatory
 467        # however, it might be a single expression
 468        body = children[-1]
 469
 470        if len(children) == 4:
 471            # handle the first trivial case - if there are four children we know exactly what they are
 472            pre, cond, post = children[0], children[1], children[2]
 473        elif len(children) == 1:
 474            # handle the second trivial case - if there is only one child we know there are no expressions
 475            pre, cond, post = None, None, None
 476        else:
 477            attributes = statement.get("attributes", None)
 478
 479            def has_hint(key):
 480                return key in attributes and not attributes[key]
 481
 482            if attributes and any(
 483                map(
 484                    has_hint,
 485                    ["condition", "initializationExpression", "loopExpression"],
 486                )
 487            ):
 488                # if we have attribute hints, rely on those
 489
 490                if len(children) == 2:
 491                    # we're missing two expressions, find the one we have
 492                    if not has_hint("initializationExpression"):
 493                        pre, cond, post = children[0], None, None
 494                    elif not has_hint("condition"):
 495                        pre, cond, post = None, children[0], None
 496                    else:  # if not has_hint('loopExpression'):
 497                        pre, cond, post = None, None, children[0]
 498                else:  # len(children) == 3
 499                    # we're missing one expression, figure out what it is
 500                    if has_hint("initializationExpression"):
 501                        pre, cond, post = None, children[0], children[1]
 502                    elif has_hint("condition"):
 503                        pre, cond, post = children[0], None, children[1]
 504                    else:  # if has_hint('loopExpression'):
 505                        pre, cond, post = children[0], children[1], None
 506            else:
 507                # we don't have attribute hints, and it's impossible to be 100% accurate here
 508                # let's just try our best
 509
 510                first_type = children[0][self.get_key()]
 511                second_type = children[1][self.get_key()]
 512
 513                # VariableDefinitionStatement is used by solc 0.4.0-0.4.6
 514                # it's changed in 0.4.7 to VariableDeclarationStatement
 515                if first_type in ["VariableDefinitionStatement", "VariableDeclarationStatement"]:
 516                    # only the pre statement can be a variable declaration
 517
 518                    if len(children) == 2:
 519                        # only one child apart from body, it must be pre
 520                        pre, cond, post = children[0], None, None
 521                    else:
 522                        # more than one child, figure out which one is the cond
 523                        if second_type == "ExpressionStatement":
 524                            # only the post can be an expression statement
 525                            pre, cond, post = children[0], None, children[1]
 526                        else:
 527                            # similarly, the post cannot be anything other than an expression statement
 528                            pre, cond, post = children[0], children[1], None
 529                elif first_type == "ExpressionStatement":
 530                    # the first element can either be pre or post
 531
 532                    if len(children) == 2:
 533                        # this is entirely ambiguous, so apply a very dumb heuristic:
 534                        # if the statement is closer to the start of the body, it's probably the post
 535                        # otherwise, it's probably the pre
 536                        # this will work in all cases where the formatting isn't completely borked
 537
 538                        node_len = int(children[0]["src"].split(":")[1])
 539
 540                        node_start = int(children[0]["src"].split(":")[0])
 541                        node_end = node_start + node_len
 542
 543                        for_start = int(statement["src"].split(":")[0]) + 3  # trim off the 'for'
 544                        body_start = int(body["src"].split(":")[0])
 545
 546                        dist_start = node_start - for_start
 547                        dist_end = body_start - node_end
 548                        if dist_start > dist_end:
 549                            pre, cond, post = None, None, children[0]
 550                        else:
 551                            pre, cond, post = children[0], None, None
 552                    else:
 553                        # more than one child, we must be the pre
 554                        pre, cond, post = children[0], children[1], None
 555                else:
 556                    # the first element must be the cond
 557
 558                    if len(children) == 2:
 559                        pre, cond, post = None, children[0], None
 560                    else:
 561                        pre, cond, post = None, children[0], children[1]
 562
 563        return pre, cond, post, body

 565    def _parse_for(self, statement: Dict, node: NodeSolc) -> NodeSolc:
 566        # ForStatement = 'for' '(' (SimpleStatement)? ';' (Expression)? ';' (ExpressionStatement)? ')' Statement
 567
 568        if self.is_compact_ast:
 569            pre, cond, post, body = self._parse_for_compact_ast(statement)
 570        else:
 571            pre, cond, post, body = self._parse_for_legacy_ast(statement)
 572
 573        node_startLoop = self._new_node(
 574            NodeType.STARTLOOP, statement["src"], node.underlying_node.scope
 575        )
 576        node_endLoop = self._new_node(
 577            NodeType.ENDLOOP, statement["src"], node.underlying_node.scope
 578        )
 579
 580        last_scope = node.underlying_node.scope
 581
 582        if pre:
 583            pre_scope = Scope(node.underlying_node.scope.is_checked, False, last_scope)
 584            last_scope = pre_scope
 585            node_init_expression = self._parse_statement(pre, node, pre_scope)
 586            link_underlying_nodes(node_init_expression, node_startLoop)
 587        else:
 588            link_underlying_nodes(node, node_startLoop)
 589
 590        if cond:
 591            cond_scope = Scope(node.underlying_node.scope.is_checked, False, last_scope)
 592            last_scope = cond_scope
 593            node_condition = self._new_node(NodeType.IFLOOP, cond["src"], cond_scope)
 594            node_condition.add_unparsed_expression(cond)
 595            link_underlying_nodes(node_startLoop, node_condition)
 596
 597            node_beforeBody = node_condition
 598        else:
 599            node_condition = None
 600            node_beforeBody = node_startLoop
 601
 602        body_scope = Scope(node.underlying_node.scope.is_checked, False, last_scope)
 603        last_scope = body_scope
 604        node_body = self._parse_statement(body, node_beforeBody, body_scope)
 605
 606        if post:
 607            node_loopexpression = self._parse_statement(post, node_body, last_scope)
 608            link_underlying_nodes(node_loopexpression, node_beforeBody)
 609        else:
 610            # node_loopexpression = None
 611            link_underlying_nodes(node_body, node_beforeBody)
 612
 613        if node_condition:
 614            link_underlying_nodes(node_condition, node_endLoop)
 615        else:
 616            link_underlying_nodes(
 617                node_startLoop, node_endLoop
 618            )  # this is an infinite loop but we can't break our cfg
 619
 620        return node_endLoop

 626        )
 627        condition_scope = Scope(
 628            node.underlying_node.scope.is_checked, False, node.underlying_node.scope
 629        )
 630
 631        if self.is_compact_ast:
 632            node_condition = self._new_node(
 633                NodeType.IFLOOP, do_while_statement["condition"]["src"], condition_scope
 634            )
 635            node_condition.add_unparsed_expression(do_while_statement["condition"])
 636            statement = self._parse_statement(
 637                do_while_statement["body"], node_condition, condition_scope
 638            )
 639        else:
 640            children = do_while_statement[self.get_children("children")]
 641            # same order in the AST as while
 642            expression = children[0]
 643            node_condition = self._new_node(NodeType.IFLOOP, expression["src"], condition_scope)
 644            node_condition.add_unparsed_expression(expression)
 645            statement = self._parse_statement(children[1], node_condition, condition_scope)
 646
 647        body_scope = Scope(node.underlying_node.scope.is_checked, False, condition_scope)
 648        node_endDoWhile = self._new_node(NodeType.ENDLOOP, do_while_statement["src"], body_scope)
 649
 650        link_underlying_nodes(node, node_startDoWhile)
 651        # empty block, loop from the start to the condition
 652        if not node_condition.underlying_node.sons:
 653            link_underlying_nodes(node_startDoWhile, node_condition)
 654        else:
 655            link_nodes(
 656                node_startDoWhile.underlying_node,
 657                node_condition.underlying_node.sons[0],
 658            )
 659        link_underlying_nodes(statement, node_condition)
 660        link_underlying_nodes(node_condition, node_endDoWhile)
 661        return node_endDoWhile

 674        node = new_node
 675
 676        for clause in statement.get("clauses", []):

 702    def _parse_variable_definition(self, statement: Dict, node: NodeSolc) -> NodeSolc:
 703        try:
 704            local_var = LocalVariable()
 705            local_var.set_function(self._function)
 706            local_var.set_offset(statement["src"], self._function.compilation_unit)
 707
 708            local_var_parser = LocalVariableSolc(local_var, statement)
 709            self._add_local_variable(local_var_parser)
 710            # local_var.analyze(self)
 711
 712            new_node = self._new_node(
 713                NodeType.VARIABLE, statement["src"], node.underlying_node.scope
 714            )
 715            new_node.underlying_node.add_variable_declaration(local_var)
 716            link_underlying_nodes(node, new_node)
 717            return new_node
 718        except MultipleVariablesDeclaration:
 719            # Custom handling of var (a,b) = .. style declaration
 720            if self.is_compact_ast:
 721                variables = statement["declarations"]
 722                count = len(variables)
 723
 724                if (
 725                    statement["initialValue"]["nodeType"] == "TupleExpression"
 726                    and len(statement["initialValue"]["components"]) == count
 727                ):
 728                    inits = statement["initialValue"]["components"]
 729                    i = 0
 730                    new_node = node
 731                    for variable in variables:
 732                        if variable is None:
 733                            continue
 734                        init = inits[i]
 735                        src = variable["src"]
 736                        i = i + 1
 737
 738                        new_statement = {
 739                            "nodeType": "VariableDefinitionStatement",
 740                            "src": src,
 741                            "declarations": [variable],
 742                            "initialValue": init,
 743                        }
 744                        new_node = self._parse_variable_definition(new_statement, new_node)
 745
 746                else:
 747                    # If we have
 748                    # var (a, b) = f()
 749                    # we can split in multiple declarations, without init
 750                    # Then we craft one expression that does the assignment
 751                    variables = []
 752                    i = 0
 753                    new_node = node
 754                    for variable in statement["declarations"]:
 755                        if variable:
 756                            src = variable["src"]
 757                            # Create a fake statement to be consistent
 758                            new_statement = {
 759                                "nodeType": "VariableDefinitionStatement",
 760                                "src": src,
 761                                "declarations": [variable],
 762                            }
 763                            variables.append(variable)
 764
 765                            new_node = self._parse_variable_definition_init_tuple(
 766                                new_statement, i, new_node
 767                            )
 768                        i = i + 1
 769
 770                    var_identifiers = []
 771                    # craft of the expression doing the assignement
 772                    for v in variables:
 773                        identifier = {
 774                            "nodeType": "Identifier",
 775                            "src": v["src"],
 776                            "name": v["name"],
 777                            "typeDescriptions": {"typeString": v["typeDescriptions"]["typeString"]},
 778                        }
 779                        var_identifiers.append(identifier)
 780
 781                    tuple_expression = {
 782                        "nodeType": "TupleExpression",
 783                        "src": statement["src"],
 784                        "components": var_identifiers,
 785                    }
 786
 787                    expression = {
 788                        "nodeType": "Assignment",
 789                        "src": statement["src"],
 790                        "operator": "=",
 791                        "type": "tuple()",
 792                        "leftHandSide": tuple_expression,
 793                        "rightHandSide": statement["initialValue"],
 794                        "typeDescriptions": {"typeString": "tuple()"},
 795                    }
 796                    node = new_node
 797                    new_node = self._new_node(
 798                        NodeType.EXPRESSION, statement["src"], node.underlying_node.scope
 799                    )
 800                    new_node.add_unparsed_expression(expression)
 801                    link_underlying_nodes(node, new_node)
 802
 803            else:
 804                count = 0
 805                children = statement[self.get_children("children")]
 806                child = children[0]
 807                while child[self.get_key()] == "VariableDeclaration":
 808                    count = count + 1
 809                    child = children[count]
 810
 811                assert len(children) == (count + 1)
 812                tuple_vars = children[count]
 813
 814                variables_declaration = children[0:count]
 815                i = 0
 816                new_node = node
 817                if tuple_vars[self.get_key()] == "TupleExpression":
 818                    assert len(tuple_vars[self.get_children("children")]) == count
 819                    for variable in variables_declaration:
 820                        init = tuple_vars[self.get_children("children")][i]
 821                        src = variable["src"]
 822                        i = i + 1
 823                        # Create a fake statement to be consistent
 824                        new_statement = {
 825                            self.get_key(): "VariableDefinitionStatement",
 826                            "src": src,
 827                            self.get_children("children"): [variable, init],
 828                        }
 829
 830                        new_node = self._parse_variable_definition(new_statement, new_node)
 831                else:
 832                    # If we have
 833                    # var (a, b) = f()
 834                    # we can split in multiple declarations, without init
 835                    # Then we craft one expression that does the assignment
 836                    assert tuple_vars[self.get_key()] in ["FunctionCall", "Conditional"]
 837                    variables = []
 838                    for variable in variables_declaration:
 839                        src = variable["src"]
 840                        # Create a fake statement to be consistent
 841                        new_statement = {
 842                            self.get_key(): "VariableDefinitionStatement",
 843                            "src": src,
 844                            self.get_children("children"): [variable],
 845                        }
 846                        variables.append(variable)
 847
 848                        new_node = self._parse_variable_definition_init_tuple(
 849                            new_statement, i, new_node
 850                        )
 851                        i = i + 1
 852                    var_identifiers = []
 853                    # craft of the expression doing the assignement
 854                    for v in variables:
 855                        identifier = {
 856                            self.get_key(): "Identifier",
 857                            "src": v["src"],
 858                            "attributes": {
 859                                "value": v["attributes"][self.get_key()],
 860                                "type": v["attributes"]["type"],
 861                            },
 862                        }
 863                        var_identifiers.append(identifier)
 864
 865                    expression = {
 866                        self.get_key(): "Assignment",
 867                        "src": statement["src"],
 868                        "attributes": {"operator": "=", "type": "tuple()"},
 869                        self.get_children("children"): [
 870                            {
 871                                self.get_key(): "TupleExpression",
 872                                "src": statement["src"],
 873                                self.get_children("children"): var_identifiers,
 874                            },
 875                            tuple_vars,
 876                        ],
 877                    }
 878                    node = new_node
 879                    new_node = self._new_node(
 880                        NodeType.EXPRESSION, statement["src"], node.underlying_node.scope
 881                    )
 882                    new_node.add_unparsed_expression(expression)
 883                    link_underlying_nodes(node, new_node)
 884
 885            return new_node

 903    def _parse_statement(
 904        self, statement: Dict, node: NodeSolc, scope: Union[Scope, Function]
 905    ) -> NodeSolc:
 906        """
 907
 908        Return:
 909            node
 910        """
 911        # Statement = IfStatement | WhileStatement | ForStatement | Block | InlineAssemblyStatement |
 912        #            ( DoWhileStatement | PlaceholderStatement | Continue | Break | Return |
 913        #                          Throw | EmitStatement | SimpleStatement ) ';'
 914        # SimpleStatement = VariableDefinition | ExpressionStatement
 915
 916        name = statement[self.get_key()]
 917        # SimpleStatement = VariableDefinition | ExpressionStatement
 918        if name == "IfStatement":
 919            node = self._parse_if(statement, node)
 920        elif name == "WhileStatement":
 921            node = self._parse_while(statement, node)
 922        elif name == "ForStatement":
 923            node = self._parse_for(statement, node)
 924        elif name == "Block":
 925            node = self._parse_block(statement, node)
 926        elif name == "UncheckedBlock":
 927            node = self._parse_unchecked_block(statement, node)
 928        elif name == "InlineAssembly":
 929            # Added with solc 0.6 - the yul code is an AST
 930            if "AST" in statement and not self.compilation_unit.core.skip_assembly:
 931                self._function.contains_assembly = True
 932                yul_object = self._new_yul_block(statement["src"], scope)
 933                entrypoint = yul_object.entrypoint
 934                exitpoint = yul_object.convert(statement["AST"])
 935
 936                # technically, entrypoint and exitpoint are YulNodes and we should be returning a NodeSolc here
 937                # but they both expose an underlying_node so oh well
 938                link_underlying_nodes(node, entrypoint)
 939                node = exitpoint
 940            else:
 941                asm_node = self._new_node(NodeType.ASSEMBLY, statement["src"], scope)
 942                self._function.contains_assembly = True
 943                # Added with solc 0.4.12
 944                if "operations" in statement:
 945                    asm_node.underlying_node.add_inline_asm(statement["operations"])
 946                link_underlying_nodes(node, asm_node)
 947                node = asm_node
 948        elif name == "DoWhileStatement":
 949            node = self._parse_dowhile(statement, node)
 950        # For Continue / Break / Return / Throw
 951        # The is fixed later
 952        elif name == "Continue":
 953            continue_node = self._new_node(NodeType.CONTINUE, statement["src"], scope)
 954            link_underlying_nodes(node, continue_node)
 955            node = continue_node
 956        elif name == "Break":
 957            break_node = self._new_node(NodeType.BREAK, statement["src"], scope)
 958            link_underlying_nodes(node, break_node)
 959            node = break_node
 960        elif name == "Return":
 961            return_node = self._new_node(NodeType.RETURN, statement["src"], scope)
 962            link_underlying_nodes(node, return_node)
 963            if self.is_compact_ast:
 964                if statement.get("expression", None):
 965                    return_node.add_unparsed_expression(statement["expression"])
 966            else:
 967                if (
 968                    self.get_children("children") in statement
 969                    and statement[self.get_children("children")]
 970                ):
 971                    assert len(statement[self.get_children("children")]) == 1
 972                    expression = statement[self.get_children("children")][0]
 973                    return_node.add_unparsed_expression(expression)
 974            node = return_node
 975        elif name == "Throw":
 976            throw_node = self._new_node(NodeType.THROW, statement["src"], scope)
 977            link_underlying_nodes(node, throw_node)
 978            node = throw_node
 979        elif name == "EmitStatement":
 980            # expression = parse_expression(statement[self.get_children('children')][0], self)
 981            if self.is_compact_ast:
 982                expression = statement["eventCall"]
 983            else:
 984                expression = statement[self.get_children("children")][0]
 985            new_node = self._new_node(NodeType.EXPRESSION, statement["src"], scope)
 986            new_node.add_unparsed_expression(expression)
 987            link_underlying_nodes(node, new_node)
 988            node = new_node
 989        elif name in ["VariableDefinitionStatement", "VariableDeclarationStatement"]:
 990            node = self._parse_variable_definition(statement, node)
 991        elif name == "ExpressionStatement":
 992            # assert len(statement[self.get_children('expression')]) == 1
 993            # assert not 'attributes' in statement
 994            # expression = parse_expression(statement[self.get_children('children')][0], self)
 995            if self.is_compact_ast:
 996                expression = statement[self.get_children("expression")]
 997            else:
 998                expression = statement[self.get_children("expression")][0]
 999            new_node = self._new_node(NodeType.EXPRESSION, statement["src"], scope)
1000            new_node.add_unparsed_expression(expression)
1001            link_underlying_nodes(node, new_node)
1002            node = new_node
1003        elif name == "TryStatement":
1004            node = self._parse_try_catch(statement, node)
1005        # elif name == 'TryCatchClause':
1006        #     self._parse_catch(statement, node)
1007        elif name == "RevertStatement":
1008            if self.is_compact_ast:
1009                expression = statement[self.get_children("errorCall")]
1010            else:
1011                expression = statement[self.get_children("errorCall")][0]
1012            new_node = self._new_node(NodeType.EXPRESSION, statement["src"], scope)
1013            new_node.add_unparsed_expression(expression)
1014            link_underlying_nodes(node, new_node)
1015            node = new_node
1016        else:
1017            raise ParsingError(f"Statement not parsed {name}")
1018
1019        return node

1084    def _find_end_loop(self, node: Node, visited: List[Node], counter: int) -> Optional[Node]:
1085        # counter allows to explore nested loop
1086        if node in visited:
1087            return None
1088
1089        if node.type == NodeType.ENDLOOP:
1090            if counter == 0:
1091                return node
1092            counter -= 1
1093
1094        # nested loop
1095        if node.type == NodeType.STARTLOOP:
1096            counter += 1
1097
1098        visited = visited + [node]
1099        for son in node.sons:
1100            ret = self._find_end_loop(son, visited, counter)
1101            if ret:
1102                return ret
1103
1104        return None

1106    def _find_start_loop(self, node: Node, visited: List[Node]) -> Optional[Node]:
1107        if node in visited:
1108            return None
1109
1110        if node.type == NodeType.STARTLOOP:
1111            return node
1112
1113        visited = visited + [node]
1114        for father in node.fathers:
1115            ret = self._find_start_loop(father, visited)
1116            if ret:
1117                return ret
1118
1119        return None

1121    def _fix_break_node(self, node: Node):
1122        end_node = self._find_end_loop(node, [], 0)
1123
1124        if not end_node:
1125            # If there is not end condition on the loop
1126            # The exploration will reach a STARTLOOP before reaching the endloop
1127            # We start with -1 as counter to catch this corner case
1128            end_node = self._find_end_loop(node, [], -1)
1129            if not end_node:
1130                raise ParsingError(f"Break in no-loop context {node.function}")
1131
1132        for son in node.sons:
1133            son.remove_father(node)
1134        node.set_sons([end_node])
1135        end_node.add_father(node)

 457    def _parse_for_legacy_ast(
 458        self, statement: Dict
 459    ) -> (Optional[Dict], Optional[Dict], Optional[Dict], Dict):
 460        # if we're using an old version of solc (anything below and including 0.4.11) or if the user
 461        # explicitly enabled compact ast, we might need to make some best-effort guesses
 462        children = statement[self.get_children("children")]
 463
 464        # there should always be at least one, and never more than 4, children
 465        assert 1 <= len(children) <= 4
 466
 467        # the last element of the children array must be the body, since it's mandatory
 468        # however, it might be a single expression
 469        body = children[-1]
 470
 471        if len(children) == 4:
 472            # handle the first trivial case - if there are four children we know exactly what they are
 473            pre, cond, post = children[0], children[1], children[2]
 474        elif len(children) == 1:
 475            # handle the second trivial case - if there is only one child we know there are no expressions
 476            pre, cond, post = None, None, None
 477        else:
 478            attributes = statement.get("attributes", None)
 479
 480            def has_hint(key):
 481                return key in attributes and not attributes[key]
 482
 483            if attributes and any(
 484                map(
 485                    has_hint,
 486                    ["condition", "initializationExpression", "loopExpression"],
 487                )
 488            ):
 489                # if we have attribute hints, rely on those
 490
 491                if len(children) == 2:
 492                    # we're missing two expressions, find the one we have
 493                    if not has_hint("initializationExpression"):
 494                        pre, cond, post = children[0], None, None
 495                    elif not has_hint("condition"):
 496                        pre, cond, post = None, children[0], None
 497                    else:  # if not has_hint('loopExpression'):
 498                        pre, cond, post = None, None, children[0]
 499                else:  # len(children) == 3
 500                    # we're missing one expression, figure out what it is
 501                    if has_hint("initializationExpression"):
 502                        pre, cond, post = None, children[0], children[1]
 503                    elif has_hint("condition"):
 504                        pre, cond, post = children[0], None, children[1]
 505                    else:  # if has_hint('loopExpression'):
 506                        pre, cond, post = children[0], children[1], None
 507            else:
 508                # we don't have attribute hints, and it's impossible to be 100% accurate here
 509                # let's just try our best
 510
 511                first_type = children[0][self.get_key()]
 512                second_type = children[1][self.get_key()]
 513
 514                # VariableDefinitionStatement is used by solc 0.4.0-0.4.6
 515                # it's changed in 0.4.7 to VariableDeclarationStatement
 516                if first_type in ["VariableDefinitionStatement", "VariableDeclarationStatement"]:
 517                    # only the pre statement can be a variable declaration
 518
 519                    if len(children) == 2:
 520                        # only one child apart from body, it must be pre
 521                        pre, cond, post = children[0], None, None
 522                    else:
 523                        # more than one child, figure out which one is the cond
 524                        if second_type == "ExpressionStatement":
 525                            # only the post can be an expression statement
 526                            pre, cond, post = children[0], None, children[1]
 527                        else:
 528                            # similarly, the post cannot be anything other than an expression statement
 529                            pre, cond, post = children[0], children[1], None
 530                elif first_type == "ExpressionStatement":
 531                    # the first element can either be pre or post
 532
 533                    if len(children) == 2:
 534                        # this is entirely ambiguous, so apply a very dumb heuristic:
 535                        # if the statement is closer to the start of the body, it's probably the post
 536                        # otherwise, it's probably the pre
 537                        # this will work in all cases where the formatting isn't completely borked
 538
 539                        node_len = int(children[0]["src"].split(":")[1])
 540
 541                        node_start = int(children[0]["src"].split(":")[0])
 542                        node_end = node_start + node_len
 543
 544                        for_start = int(statement["src"].split(":")[0]) + 3  # trim off the 'for'
 545                        body_start = int(body["src"].split(":")[0])
 546
 547                        dist_start = node_start - for_start
 548                        dist_end = body_start - node_end
 549                        if dist_start > dist_end:
 550                            pre, cond, post = None, None, children[0]
 551                        else:
 552                            pre, cond, post = children[0], None, None
 553                    else:
 554                        # more than one child, we must be the pre
 555                        pre, cond, post = children[0], children[1], None
 556                else:
 557                    # the first element must be the cond
 558
 559                    if len(children) == 2:
 560                        pre, cond, post = None, children[0], None
 561                    else:
 562                        pre, cond, post = None, children[0], children[1]
 563
 564        return pre, cond, post, body

 566    def _parse_for(self, statement: Dict, node: NodeSolc) -> NodeSolc:
 567        # ForStatement = 'for' '(' (SimpleStatement)? ';' (Expression)? ';' (ExpressionStatement)? ')' Statement
 568
 569        if self.is_compact_ast:
 570            pre, cond, post, body = self._parse_for_compact_ast(statement)
 571        else:
 572            pre, cond, post, body = self._parse_for_legacy_ast(statement)
 573
 574        node_startLoop = self._new_node(
 575            NodeType.STARTLOOP, statement["src"], node.underlying_node.scope
 576        )
 577        node_endLoop = self._new_node(
 578            NodeType.ENDLOOP, statement["src"], node.underlying_node.scope
 579        )
 580
 581        last_scope = node.underlying_node.scope
 582
 583        if pre:
 584            pre_scope = Scope(node.underlying_node.scope.is_checked, False, last_scope)
 585            last_scope = pre_scope
 586            node_init_expression = self._parse_statement(pre, node, pre_scope)
 587            link_underlying_nodes(node_init_expression, node_startLoop)
 588        else:
 589            link_underlying_nodes(node, node_startLoop)
 590
 591        if cond:
 592            cond_scope = Scope(node.underlying_node.scope.is_checked, False, last_scope)
 593            last_scope = cond_scope
 594            node_condition = self._new_node(NodeType.IFLOOP, cond["src"], cond_scope)
 595            node_condition.add_unparsed_expression(cond)
 596            link_underlying_nodes(node_startLoop, node_condition)
 597
 598            node_beforeBody = node_condition
 599        else:
 600            node_condition = None
 601            node_beforeBody = node_startLoop
 602
 603        body_scope = Scope(node.underlying_node.scope.is_checked, False, last_scope)
 604        last_scope = body_scope
 605        node_body = self._parse_statement(body, node_beforeBody, body_scope)
 606
 607        if post:
 608            node_loopexpression = self._parse_statement(post, node_body, last_scope)
 609            link_underlying_nodes(node_loopexpression, node_beforeBody)
 610        else:
 611            # node_loopexpression = None
 612            link_underlying_nodes(node_body, node_beforeBody)
 613
 614        if node_condition:
 615            link_underlying_nodes(node_condition, node_endLoop)
 616        else:
 617            link_underlying_nodes(
 618                node_startLoop, node_endLoop
 619            )  # this is an infinite loop but we can't break our cfg
 620
 621        return node_endLoop

 623    def _parse_dowhile(self, do_while_statement: Dict, node: NodeSolc) -> NodeSolc:
 624
 625        node_startDoWhile = self._new_node(
 626            NodeType.STARTLOOP, do_while_statement["src"], node.underlying_node.scope
 627        )
 628        condition_scope = Scope(
 629            node.underlying_node.scope.is_checked, False, node.underlying_node.scope
 630        )
 631
 632        if self.is_compact_ast:
 633            node_condition = self._new_node(
 634                NodeType.IFLOOP, do_while_statement["condition"]["src"], condition_scope
 635            )
 636            node_condition.add_unparsed_expression(do_while_statement["condition"])
 637            statement = self._parse_statement(
 638                do_while_statement["body"], node_condition, condition_scope
 639            )
 640        else:
 641            children = do_while_statement[self.get_children("children")]
 642            # same order in the AST as while
 643            expression = children[0]
 644            node_condition = self._new_node(NodeType.IFLOOP, expression["src"], condition_scope)
 645            node_condition.add_unparsed_expression(expression)
 646            statement = self._parse_statement(children[1], node_condition, condition_scope)
 647
 648        body_scope = Scope(node.underlying_node.scope.is_checked, False, condition_scope)
 649        node_endDoWhile = self._new_node(NodeType.ENDLOOP, do_while_statement["src"], body_scope)
 650
 651        link_underlying_nodes(node, node_startDoWhile)
 652        # empty block, loop from the start to the condition
 653        if not node_condition.underlying_node.sons:
 654            link_underlying_nodes(node_startDoWhile, node_condition)
 655        else:
 656            link_nodes(
 657                node_startDoWhile.underlying_node,
 658                node_condition.underlying_node.sons[0],
 659            )
 660        link_underlying_nodes(statement, node_condition)
 661        link_underlying_nodes(node_condition, node_endDoWhile)
 662        return node_endDoWhile

 664    def _parse_try_catch(self, statement: Dict, node: NodeSolc) -> NodeSolc:
 665        externalCall = statement.get("externalCall", None)
 666
 667        if externalCall is None:
 668            raise ParsingError(f"Try/Catch not correctly parsed by Slither {statement}")
 669        catch_scope = Scope(
 670            node.underlying_node.scope.is_checked, False, node.underlying_node.scope
 671        )
 672        new_node = self._new_node(NodeType.TRY, statement["src"], catch_scope)
 673        new_node.add_unparsed_expression(externalCall)
 674        link_underlying_nodes(node, new_node)
 675        node = new_node
 676
 677        for clause in statement.get("clauses", []):
 678            self._parse_catch(clause, node)
 679        return node

 681    def _parse_catch(self, statement: Dict, node: NodeSolc) -> NodeSolc:
 682        block = statement.get("block", None)
 683
 684        if block is None:
 685            raise ParsingError(f"Catch not correctly parsed by Slither {statement}")
 686        try_scope = Scope(node.underlying_node.scope.is_checked, False, node.underlying_node.scope)
 687
 688        try_node = self._new_node(NodeType.CATCH, statement["src"], try_scope)
 689        link_underlying_nodes(node, try_node)
 690
 691        if self.is_compact_ast:
 692            params = statement.get("parameters", None)
 693        else:
 694            params = statement[self.get_children("children")]
 695
 696        if params:
 697            for param in params.get("parameters", []):
 698                assert param[self.get_key()] == "VariableDeclaration"
 699                self._add_param(param)
 700
 701        return self._parse_statement(block, try_node, try_scope)

 703    def _parse_variable_definition(self, statement: Dict, node: NodeSolc) -> NodeSolc:
 704        try:
 705            local_var = LocalVariable()
 706            local_var.set_function(self._function)
 707            local_var.set_offset(statement["src"], self._function.compilation_unit)
 708
 709            local_var_parser = LocalVariableSolc(local_var, statement)
 710            self._add_local_variable(local_var_parser)
 711            # local_var.analyze(self)
 712
 713            new_node = self._new_node(
 714                NodeType.VARIABLE, statement["src"], node.underlying_node.scope
 715            )
 716            new_node.underlying_node.add_variable_declaration(local_var)
 717            link_underlying_nodes(node, new_node)
 718            return new_node
 719        except MultipleVariablesDeclaration:
 720            # Custom handling of var (a,b) = .. style declaration
 721            if self.is_compact_ast:
 722                variables = statement["declarations"]
 723                count = len(variables)
 724
 725                if (
 726                    statement["initialValue"]["nodeType"] == "TupleExpression"
 727                    and len(statement["initialValue"]["components"]) == count
 728                ):
 729                    inits = statement["initialValue"]["components"]
 730                    i = 0
 731                    new_node = node
 732                    for variable in variables:
 733                        if variable is None:
 734                            continue
 735                        init = inits[i]
 736                        src = variable["src"]
 737                        i = i + 1
 738
 739                        new_statement = {
 740                            "nodeType": "VariableDefinitionStatement",
 741                            "src": src,
 742                            "declarations": [variable],
 743                            "initialValue": init,
 744                        }
 745                        new_node = self._parse_variable_definition(new_statement, new_node)
 746
 747                else:
 748                    # If we have
 749                    # var (a, b) = f()
 750                    # we can split in multiple declarations, without init
 751                    # Then we craft one expression that does the assignment
 752                    variables = []
 753                    i = 0
 754                    new_node = node
 755                    for variable in statement["declarations"]:
 756                        if variable:
 757                            src = variable["src"]
 758                            # Create a fake statement to be consistent
 759                            new_statement = {
 760                                "nodeType": "VariableDefinitionStatement",
 761                                "src": src,
 762                                "declarations": [variable],
 763                            }
 764                            variables.append(variable)
 765
 766                            new_node = self._parse_variable_definition_init_tuple(
 767                                new_statement, i, new_node
 768                            )
 769                        i = i + 1
 770
 771                    var_identifiers = []
 772                    # craft of the expression doing the assignement
 773                    for v in variables:
 774                        identifier = {
 775                            "nodeType": "Identifier",
 776                            "src": v["src"],
 777                            "name": v["name"],
 778                            "typeDescriptions": {"typeString": v["typeDescriptions"]["typeString"]},
 779                        }
 780                        var_identifiers.append(identifier)
 781
 782                    tuple_expression = {
 783                        "nodeType": "TupleExpression",
 784                        "src": statement["src"],
 785                        "components": var_identifiers,
 786                    }
 787
 788                    expression = {
 789                        "nodeType": "Assignment",
 790                        "src": statement["src"],
 791                        "operator": "=",
 792                        "type": "tuple()",
 793                        "leftHandSide": tuple_expression,
 794                        "rightHandSide": statement["initialValue"],
 795                        "typeDescriptions": {"typeString": "tuple()"},
 796                    }
 797                    node = new_node
 798                    new_node = self._new_node(
 799                        NodeType.EXPRESSION, statement["src"], node.underlying_node.scope
 800                    )
 801                    new_node.add_unparsed_expression(expression)
 802                    link_underlying_nodes(node, new_node)
 803
 804            else:
 805                count = 0
 806                children = statement[self.get_children("children")]
 807                child = children[0]
 808                while child[self.get_key()] == "VariableDeclaration":
 809                    count = count + 1
 810                    child = children[count]
 811
 812                assert len(children) == (count + 1)
 813                tuple_vars = children[count]
 814
 815                variables_declaration = children[0:count]
 816                i = 0
 817                new_node = node
 818                if tuple_vars[self.get_key()] == "TupleExpression":
 819                    assert len(tuple_vars[self.get_children("children")]) == count
 820                    for variable in variables_declaration:
 821                        init = tuple_vars[self.get_children("children")][i]
 822                        src = variable["src"]
 823                        i = i + 1
 824                        # Create a fake statement to be consistent
 825                        new_statement = {
 826                            self.get_key(): "VariableDefinitionStatement",
 827                            "src": src,
 828                            self.get_children("children"): [variable, init],
 829                        }
 830
 831                        new_node = self._parse_variable_definition(new_statement, new_node)
 832                else:
 833                    # If we have
 834                    # var (a, b) = f()
 835                    # we can split in multiple declarations, without init
 836                    # Then we craft one expression that does the assignment
 837                    assert tuple_vars[self.get_key()] in ["FunctionCall", "Conditional"]
 838                    variables = []
 839                    for variable in variables_declaration:
 840                        src = variable["src"]
 841                        # Create a fake statement to be consistent
 842                        new_statement = {
 843                            self.get_key(): "VariableDefinitionStatement",
 844                            "src": src,
 845                            self.get_children("children"): [variable],
 846                        }
 847                        variables.append(variable)
 848
 849                        new_node = self._parse_variable_definition_init_tuple(
 850                            new_statement, i, new_node
 851                        )
 852                        i = i + 1
 853                    var_identifiers = []
 854                    # craft of the expression doing the assignement
 855                    for v in variables:
 856                        identifier = {
 857                            self.get_key(): "Identifier",
 858                            "src": v["src"],
 859                            "attributes": {
 860                                "value": v["attributes"][self.get_key()],
 861                                "type": v["attributes"]["type"],
 862                            },
 863                        }
 864                        var_identifiers.append(identifier)
 865
 866                    expression = {
 867                        self.get_key(): "Assignment",
 868                        "src": statement["src"],
 869                        "attributes": {"operator": "=", "type": "tuple()"},
 870                        self.get_children("children"): [
 871                            {
 872                                self.get_key(): "TupleExpression",
 873                                "src": statement["src"],
 874                                self.get_children("children"): var_identifiers,
 875                            },
 876                            tuple_vars,
 877                        ],
 878                    }
 879                    node = new_node
 880                    new_node = self._new_node(
 881                        NodeType.EXPRESSION, statement["src"], node.underlying_node.scope
 882                    )
 883                    new_node.add_unparsed_expression(expression)
 884                    link_underlying_nodes(node, new_node)
 885
 886            return new_node

 904    def _parse_statement(
 905        self, statement: Dict, node: NodeSolc, scope: Union[Scope, Function]
 906    ) -> NodeSolc:
 907        """
 908
 909        Return:
 910            node
 911        """
 912        # Statement = IfStatement | WhileStatement | ForStatement | Block | InlineAssemblyStatement |
 913        #            ( DoWhileStatement | PlaceholderStatement | Continue | Break | Return |
 914        #                          Throw | EmitStatement | SimpleStatement ) ';'
 915        # SimpleStatement = VariableDefinition | ExpressionStatement
 916
 917        name = statement[self.get_key()]
 918        # SimpleStatement = VariableDefinition | ExpressionStatement
 919        if name == "IfStatement":
 920            node = self._parse_if(statement, node)
 921        elif name == "WhileStatement":
 922            node = self._parse_while(statement, node)
 923        elif name == "ForStatement":
 924            node = self._parse_for(statement, node)
 925        elif name == "Block":
 926            node = self._parse_block(statement, node)
 927        elif name == "UncheckedBlock":
 928            node = self._parse_unchecked_block(statement, node)
 929        elif name == "InlineAssembly":
 930            # Added with solc 0.6 - the yul code is an AST
 931            if "AST" in statement and not self.compilation_unit.core.skip_assembly:
 932                self._function.contains_assembly = True
 933                yul_object = self._new_yul_block(statement["src"], scope)
 934                entrypoint = yul_object.entrypoint
 935                exitpoint = yul_object.convert(statement["AST"])
 936
 937                # technically, entrypoint and exitpoint are YulNodes and we should be returning a NodeSolc here
 938                # but they both expose an underlying_node so oh well
 939                link_underlying_nodes(node, entrypoint)
 940                node = exitpoint
 941            else:
 942                asm_node = self._new_node(NodeType.ASSEMBLY, statement["src"], scope)
 943                self._function.contains_assembly = True
 944                # Added with solc 0.4.12
 945                if "operations" in statement:
 946                    asm_node.underlying_node.add_inline_asm(statement["operations"])
 947                link_underlying_nodes(node, asm_node)
 948                node = asm_node
 949        elif name == "DoWhileStatement":
 950            node = self._parse_dowhile(statement, node)
 951        # For Continue / Break / Return / Throw
 952        # The is fixed later
 953        elif name == "Continue":
 954            continue_node = self._new_node(NodeType.CONTINUE, statement["src"], scope)
 955            link_underlying_nodes(node, continue_node)
 956            node = continue_node
 957        elif name == "Break":
 958            break_node = self._new_node(NodeType.BREAK, statement["src"], scope)
 959            link_underlying_nodes(node, break_node)
 960            node = break_node
 961        elif name == "Return":
 962            return_node = self._new_node(NodeType.RETURN, statement["src"], scope)
 963            link_underlying_nodes(node, return_node)
 964            if self.is_compact_ast:
 965                if statement.get("expression", None):
 966                    return_node.add_unparsed_expression(statement["expression"])
 967            else:
 968                if (
 969                    self.get_children("children") in statement
 970                    and statement[self.get_children("children")]
 971                ):
 972                    assert len(statement[self.get_children("children")]) == 1
 973                    expression = statement[self.get_children("children")][0]
 974                    return_node.add_unparsed_expression(expression)
 975            node = return_node
 976        elif name == "Throw":
 977            throw_node = self._new_node(NodeType.THROW, statement["src"], scope)
 978            link_underlying_nodes(node, throw_node)
 979            node = throw_node
 980        elif name == "EmitStatement":
 981            # expression = parse_expression(statement[self.get_children('children')][0], self)
 982            if self.is_compact_ast:
 983                expression = statement["eventCall"]
 984            else:
 985                expression = statement[self.get_children("children")][0]
 986            new_node = self._new_node(NodeType.EXPRESSION, statement["src"], scope)
 987            new_node.add_unparsed_expression(expression)
 988            link_underlying_nodes(node, new_node)
 989            node = new_node
 990        elif name in ["VariableDefinitionStatement", "VariableDeclarationStatement"]:
 991            node = self._parse_variable_definition(statement, node)
 992        elif name == "ExpressionStatement":
 993            # assert len(statement[self.get_children('expression')]) == 1
 994            # assert not 'attributes' in statement
 995            # expression = parse_expression(statement[self.get_children('children')][0], self)
 996            if self.is_compact_ast:
 997                expression = statement[self.get_children("expression")]
 998            else:
 999                expression = statement[self.get_children("expression")][0]
1000            new_node = self._new_node(NodeType.EXPRESSION, statement["src"], scope)
1001            new_node.add_unparsed_expression(expression)
1002            link_underlying_nodes(node, new_node)
1003            node = new_node
1004        elif name == "TryStatement":
1005            node = self._parse_try_catch(statement, node)
1006        # elif name == 'TryCatchClause':
1007        #     self._parse_catch(statement, node)
1008        elif name == "RevertStatement":
1009            if self.is_compact_ast:
1010                expression = statement[self.get_children("errorCall")]
1011            else:
1012                expression = statement[self.get_children("errorCall")][0]
1013            new_node = self._new_node(NodeType.EXPRESSION, statement["src"], scope)
1014            new_node.add_unparsed_expression(expression)
1015            link_underlying_nodes(node, new_node)
1016            node = new_node
1017        else:
1018            raise ParsingError(f"Statement not parsed {name}")
1019
1020        return node

1022    def _parse_block(self, block: Dict, node: NodeSolc, check_arithmetic: bool = False):
1023        """
1024        Return:
1025            Node
1026        """
1027        assert block[self.get_key()] == "Block"
1028

1031        else:

1033

1035        new_scope = Scope(check_arithmetic, False, node.underlying_node.scope)
1036        for statement in statements:
1037            node = self._parse_statement(statement, node, new_scope)
1038        return node

1085    def _find_end_loop(self, node: Node, visited: List[Node], counter: int) -> Optional[Node]:
1086        # counter allows to explore nested loop
1087        if node in visited:
1088            return None
