181    def _parse_base_contract_info(self):  # pylint: disable=too-many-branches
182        # Parse base contracts (immediate, non-linearized)
183        if self.is_compact_ast:
184            # Parse base contracts + constructors in compact-ast
185            if "baseContracts" in self._data:
186                for base_contract in self._data["baseContracts"]:
187                    if base_contract["nodeType"] != "InheritanceSpecifier":
188                        continue
189                    if (
190                        "baseName" not in base_contract
191                        or "referencedDeclaration" not in base_contract["baseName"]
192                    ):
193                        continue
194
195                    # Obtain our contract reference and add it to our base contract list
196                    referencedDeclaration = base_contract["baseName"]["referencedDeclaration"]
197                    self.baseContracts.append(referencedDeclaration)
198
199                    # If we have defined arguments in our arguments object, this is a constructor invocation.
200                    # (note: 'arguments' can be [], which is not the same as None. [] implies a constructor was
201                    #  called with no arguments, while None implies no constructor was called).
202                    if "arguments" in base_contract and base_contract["arguments"] is not None:
203                        self.baseConstructorContractsCalled.append(referencedDeclaration)
204        else:
205            # Parse base contracts + constructors in legacy-ast
206            if "children" in self._data:
207                for base_contract in self._data["children"]:
208                    if base_contract["name"] != "InheritanceSpecifier":
209                        continue
210                    if "children" not in base_contract or len(base_contract["children"]) == 0:
211                        continue
212                    # Obtain all items for this base contract specification (base contract, followed by arguments)
213                    base_contract_items = base_contract["children"]
214                    if (
215                        "name" not in base_contract_items[0]
216                        or base_contract_items[0]["name"] != "UserDefinedTypeName"
217                    ):
218                        continue
219                    if (
220                        "attributes" not in base_contract_items[0]
221                        or "referencedDeclaration" not in base_contract_items[0]["attributes"]
222                    ):
223                        continue
224
225                    # Obtain our contract reference and add it to our base contract list
226                    referencedDeclaration = base_contract_items[0]["attributes"][
227                        "referencedDeclaration"
228                    ]
229                    self.baseContracts.append(referencedDeclaration)
230
231                    # If we have an 'attributes'->'arguments' which is None, this is not a constructor call.
232                    if (
233                        "attributes" not in base_contract
234                        or "arguments" not in base_contract["attributes"]
235                        or base_contract["attributes"]["arguments"] is not None
236                    ):
237                        self.baseConstructorContractsCalled.append(referencedDeclaration)

239    def _parse_contract_items(self):
240        # pylint: disable=too-many-branches
241        if not self.get_children() in self._data:  # empty contract
242            return
243        for item in self._data[self.get_children()]:
244            if item[self.get_key()] == "FunctionDefinition":
245                self._functionsNotParsed.append(item)
246            elif item[self.get_key()] == "EventDefinition":
247                self._eventsNotParsed.append(item)
248            elif item[self.get_key()] == "InheritanceSpecifier":
249                # we dont need to parse it as it is redundant
250                # with self.linearizedBaseContracts
251                continue
252            elif item[self.get_key()] == "VariableDeclaration":
253                self._variablesNotParsed.append(item)
254            elif item[self.get_key()] == "EnumDefinition":
255                self._enumsNotParsed.append(item)
256            elif item[self.get_key()] == "ModifierDefinition":
257                self._modifiersNotParsed.append(item)
258            elif item[self.get_key()] == "StructDefinition":
259                self._structuresNotParsed.append(item)
260            elif item[self.get_key()] == "UsingForDirective":
261                self._usingForNotParsed.append(item)
262            elif item[self.get_key()] == "ErrorDefinition":
263                self._customErrorParsed.append(item)
264            elif item[self.get_key()] == "UserDefinedValueTypeDefinition":
265                self._parse_type_alias(item)
266            else:
267                raise ParsingError("Unknown contract item: " + item[self.get_key()])
268        return

270    def _parse_type_alias(self, item: Dict) -> None:
271        assert "name" in item
272        assert "underlyingType" in item
273        underlying_type = item["underlyingType"]
274        assert "nodeType" in underlying_type and underlying_type["nodeType"] == "ElementaryTypeName"
275        assert "name" in underlying_type
276
277        original_type = ElementaryType(underlying_type["name"])
278
279        # For user defined types defined at the contract level the lookup can be done
280        # Using the name or the canonical name
281        # For example during the type parsing the canonical name
282        # Note that Solidity allows shadowing of user defined types
283        # Between top level and contract definitions
284        name = item["name"]
285        canonical_name = self._contract.name + "." + item["name"]
286
287        user_defined_type = TypeAliasContract(original_type, name, self.underlying_contract)
288        user_defined_type.set_offset(item["src"], self.compilation_unit)
289        self._contract.file_scope.user_defined_types[name] = user_defined_type
290        self._contract.file_scope.user_defined_types[canonical_name] = user_defined_type
291        self._contract.file_scope.symbol_map[name + item["src"]] = DeclarationType.TypeAlias

461    def _analyze_params_element(  # pylint: disable=too-many-arguments
462        self,
463        Cls: Callable,
464        Cls_parser: Callable,
465        element_parser: FunctionSolc,
466        explored_reference_id: Set[str],
467        parser: List[FunctionSolc],
468        all_elements: Dict[str, Function],
469    ):
470        elem = Cls(self._contract.compilation_unit)
471        elem.set_contract(self._contract)
472        underlying_function = element_parser.underlying_function
473        # TopLevel function are not analyzed here
474        assert isinstance(underlying_function, FunctionContract)
475        elem.set_contract_declarer(underlying_function.contract_declarer)
476        elem.set_offset(
477            element_parser.function_not_parsed["src"],
478            self._contract.compilation_unit,
479        )
480
481        elem_parser = Cls_parser(
482            elem, element_parser.function_not_parsed, self, self.slither_parser
483        )
484        if (
485            element_parser.underlying_function.id
486            and element_parser.underlying_function.id in explored_reference_id
487        ):
488            # Already added from other fathers
489            return
490        if element_parser.underlying_function.id:
491            explored_reference_id.add(element_parser.underlying_function.id)
492        elem_parser.analyze_params()
493        if isinstance(elem, Modifier):
494            self._contract.compilation_unit.add_modifier(elem)
495        else:
496            self._contract.compilation_unit.add_function(elem)
497
498        self._slither_parser.add_function_or_modifier_parser(elem_parser)
499
500        all_elements[elem.canonical_name] = elem
501        parser.append(elem_parser)

503    def _analyze_params_elements(  # pylint: disable=too-many-arguments,too-many-locals
504        self,
505        elements_no_params: List[FunctionSolc],
506        getter: Callable[["ContractSolc"], List[FunctionSolc]],
507        getter_available: Callable[[Contract], List[FunctionContract]],
508        Cls: Callable,
509        Cls_parser: Callable,
510        parser: List[FunctionSolc],
511    ) -> Dict[str, Union[FunctionContract, Modifier]]:
512        """
513        Analyze the parameters of the given elements (Function or Modifier).
514        The function iterates over the inheritance to create an instance or inherited elements (Function or Modifier)
515        If the element is shadowed, set is_shadowed to True
516
517        :param elements_no_params: list of elements to analyzer
518        :param getter: fun x
519        :param getter_available: fun x
520        :param Cls: Class to create for collision
521        :return:
522        """
523        all_elements = {}
524
525        explored_reference_id = set()
526        try:
527            for father in self._contract.inheritance:
528                father_parser = self._slither_parser.underlying_contract_to_parser[father]
529                for element_parser in getter(father_parser):
530                    self._analyze_params_element(
531                        Cls, Cls_parser, element_parser, explored_reference_id, parser, all_elements
532                    )
533
534            accessible_elements = self._contract.available_elements_from_inheritances(
535                all_elements, getter_available
536            )
537
538            # If there is a constructor in the functions
539            # We remove the previous constructor
540            # As only one constructor is present per contracts
541            #
542            # Note: contract.all_functions_called returns the constructors of the base contracts
543            has_constructor = False
544            for element_parser in elements_no_params:
545                element_parser.analyze_params()
546                if element_parser.underlying_function.is_constructor:
547                    has_constructor = True
548
549            if has_constructor:
550                _accessible_functions = {
551                    k: v for (k, v) in accessible_elements.items() if not v.is_constructor
552                }
553
554            for element_parser in elements_no_params:
555                accessible_elements[
556                    element_parser.underlying_function.full_name
557                ] = element_parser.underlying_function
558                all_elements[
559                    element_parser.underlying_function.canonical_name
560                ] = element_parser.underlying_function
561
562            for element in all_elements.values():
563                if accessible_elements[element.full_name] != all_elements[element.canonical_name]:
564                    element.is_shadowed = True
565                    accessible_elements[element.full_name].shadows = True
566        except (VariableNotFound, KeyError) as e:
567            self.log_incorrect_parsing(f"Missing params {e}")
568        return all_elements

587    def analyze_using_for(self):  # pylint: disable=too-many-branches
588        try:
589            for father in self._contract.inheritance:
590                self._contract.using_for.update(father.using_for)
591            if self.is_compact_ast:
592                for using_for in self._usingForNotParsed:
593                    if "typeName" in using_for and using_for["typeName"]:
594                        type_name = parse_type(using_for["typeName"], self)
595                    else:
596                        type_name = "*"
597                    if type_name not in self._contract.using_for:
598                        self._contract.using_for[type_name] = []
599
600                    if "libraryName" in using_for:
601                        self._contract.using_for[type_name].append(
602                            parse_type(using_for["libraryName"], self)
603                        )
604                    else:
605                        # We have a list of functions. A function can be topLevel or a library function
606                        self._analyze_function_list(using_for["functionList"], type_name)
607            else:
608                for using_for in self._usingForNotParsed:
609                    children = using_for[self.get_children()]
610                    assert children and len(children) <= 2
611                    if len(children) == 2:
612                        new = parse_type(children[0], self)
613                        old = parse_type(children[1], self)
614                    else:
615                        new = parse_type(children[0], self)
616                        old = "*"
617                    if old not in self._contract.using_for:
618                        self._contract.using_for[old] = []
619                    self._contract.using_for[old].append(new)
620            self._usingForNotParsed = []
621        except (VariableNotFound, KeyError) as e:
622            self.log_incorrect_parsing(f"Missing using for {e}")
623
624    def _analyze_function_list(self, function_list: List, type_name: Type):
625        for f in function_list:
626            full_name_split = f["function"]["name"].split(".")
627            if len(full_name_split) == 1:
628                # Top level function
629                function_name = full_name_split[0]
630                self._analyze_top_level_function(function_name, type_name)
631            elif len(full_name_split) == 2:
632                # It can be a top level function behind an aliased import
633                # or a library function
634                first_part = full_name_split[0]
635                function_name = full_name_split[1]
636                self._check_aliased_import(first_part, function_name, type_name)
637            else:
638                # MyImport.MyLib.a we don't care of the alias
639                library_name = full_name_split[1]
640                function_name = full_name_split[2]
641                self._analyze_library_function(library_name, function_name, type_name)
642
643    def _check_aliased_import(self, first_part: str, function_name: str, type_name: Type):
644        # We check if the first part appear as alias for an import
645        # if it is then function_name must be a top level function
646        # otherwise it's a library function
647        is_import = first_part in self._contract.file_scope.imports
648        if is_import:
649            self._analyze_top_level_function(function_name, type_name)
650            return
651        self._analyze_library_function(first_part, function_name, type_name)
652
653    def _analyze_top_level_function(self, function_name: str, type_name: Type):
654        tl_function = self.compilation_unit.functions_top_level.get(function_name, None)
655        if tl_function:
656            self._contract.using_for[type_name].append(tl_function)
657
658    def _analyze_library_function(
659        self, library_name: str, function_name: str, type_name: Type
660    ) -> None:
661        # Get the library function
662        found = False
663        for c in self.compilation_unit.contracts:
664            if found:
665                break
666            if c.name == library_name:
667                for f in c.functions:
668                    if f.name == function_name:
669                        self._contract.using_for[type_name].append(f)
670                        found = True
671                        break
672        if not found:
673            self.log_incorrect_parsing(
674                f"Contract level using for: Library {library_name} - function {function_name} not found"
675            )

679            for father in self._contract.inheritance:

711        new_enum.set_offset(enum["src"], self._contract.compilation_unit)
712        self._contract.enums_as_dict[canonicalName] = new_enum

182    def _parse_base_contract_info(self):  # pylint: disable=too-many-branches
183        # Parse base contracts (immediate, non-linearized)
184        if self.is_compact_ast:
185            # Parse base contracts + constructors in compact-ast
186            if "baseContracts" in self._data:
187                for base_contract in self._data["baseContracts"]:
188                    if base_contract["nodeType"] != "InheritanceSpecifier":
189                        continue
190                    if (
191                        "baseName" not in base_contract
192                        or "referencedDeclaration" not in base_contract["baseName"]
193                    ):
194                        continue
195
196                    # Obtain our contract reference and add it to our base contract list
197                    referencedDeclaration = base_contract["baseName"]["referencedDeclaration"]
198                    self.baseContracts.append(referencedDeclaration)
199
200                    # If we have defined arguments in our arguments object, this is a constructor invocation.
201                    # (note: 'arguments' can be [], which is not the same as None. [] implies a constructor was
202                    #  called with no arguments, while None implies no constructor was called).
203                    if "arguments" in base_contract and base_contract["arguments"] is not None:
204                        self.baseConstructorContractsCalled.append(referencedDeclaration)
205        else:
206            # Parse base contracts + constructors in legacy-ast
207            if "children" in self._data:
208                for base_contract in self._data["children"]:
209                    if base_contract["name"] != "InheritanceSpecifier":
210                        continue
211                    if "children" not in base_contract or len(base_contract["children"]) == 0:
212                        continue
213                    # Obtain all items for this base contract specification (base contract, followed by arguments)
214                    base_contract_items = base_contract["children"]
215                    if (
216                        "name" not in base_contract_items[0]
217                        or base_contract_items[0]["name"] != "UserDefinedTypeName"
218                    ):
219                        continue
220                    if (
221                        "attributes" not in base_contract_items[0]
222                        or "referencedDeclaration" not in base_contract_items[0]["attributes"]
223                    ):
224                        continue
225
226                    # Obtain our contract reference and add it to our base contract list
227                    referencedDeclaration = base_contract_items[0]["attributes"][
228                        "referencedDeclaration"
229                    ]
230                    self.baseContracts.append(referencedDeclaration)
231
232                    # If we have an 'attributes'->'arguments' which is None, this is not a constructor call.
233                    if (
234                        "attributes" not in base_contract
235                        or "arguments" not in base_contract["attributes"]
236                        or base_contract["attributes"]["arguments"] is not None
237                    ):
238                        self.baseConstructorContractsCalled.append(referencedDeclaration)

240    def _parse_contract_items(self):
241        # pylint: disable=too-many-branches
242        if not self.get_children() in self._data:  # empty contract
243            return
244        for item in self._data[self.get_children()]:
245            if item[self.get_key()] == "FunctionDefinition":
246                self._functionsNotParsed.append(item)
247            elif item[self.get_key()] == "EventDefinition":
248                self._eventsNotParsed.append(item)
249            elif item[self.get_key()] == "InheritanceSpecifier":
250                # we dont need to parse it as it is redundant
251                # with self.linearizedBaseContracts
252                continue
253            elif item[self.get_key()] == "VariableDeclaration":
254                self._variablesNotParsed.append(item)
255            elif item[self.get_key()] == "EnumDefinition":
256                self._enumsNotParsed.append(item)
257            elif item[self.get_key()] == "ModifierDefinition":
258                self._modifiersNotParsed.append(item)
259            elif item[self.get_key()] == "StructDefinition":
260                self._structuresNotParsed.append(item)
261            elif item[self.get_key()] == "UsingForDirective":
262                self._usingForNotParsed.append(item)
263            elif item[self.get_key()] == "ErrorDefinition":
264                self._customErrorParsed.append(item)
265            elif item[self.get_key()] == "UserDefinedValueTypeDefinition":
266                self._parse_type_alias(item)
267            else:
268                raise ParsingError("Unknown contract item: " + item[self.get_key()])
269        return

271    def _parse_type_alias(self, item: Dict) -> None:
272        assert "name" in item
273        assert "underlyingType" in item
274        underlying_type = item["underlyingType"]
275        assert "nodeType" in underlying_type and underlying_type["nodeType"] == "ElementaryTypeName"
276        assert "name" in underlying_type
277
278        original_type = ElementaryType(underlying_type["name"])
279
280        # For user defined types defined at the contract level the lookup can be done
281        # Using the name or the canonical name
282        # For example during the type parsing the canonical name
283        # Note that Solidity allows shadowing of user defined types
284        # Between top level and contract definitions
285        name = item["name"]
286        canonical_name = self._contract.name + "." + item["name"]
287
288        user_defined_type = TypeAliasContract(original_type, name, self.underlying_contract)
289        user_defined_type.set_offset(item["src"], self.compilation_unit)
290        self._contract.file_scope.user_defined_types[name] = user_defined_type
291        self._contract.file_scope.user_defined_types[canonical_name] = user_defined_type
292        self._contract.file_scope.symbol_map[name + item["src"]] = DeclarationType.TypeAlias

625    def _analyze_function_list(self, function_list: List, type_name: Type):
626        for f in function_list:

628            if len(full_name_split) == 1:
629                # Top level function
630                function_name = full_name_split[0]
631                self._analyze_top_level_function(function_name, type_name)
632            elif len(full_name_split) == 2:
633                # It can be a top level function behind an aliased import
634                # or a library function
635                first_part = full_name_split[0]
636                function_name = full_name_split[1]
637                self._check_aliased_import(first_part, function_name, type_name)
638            else:
639                # MyImport.MyLib.a we don't care of the alias
640                library_name = full_name_split[1]
641                function_name = full_name_split[2]
642                self._analyze_library_function(library_name, function_name, type_name)
