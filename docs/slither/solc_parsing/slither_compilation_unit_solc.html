 37def _handle_import_aliases(
 38    symbol_aliases: Dict, import_directive: Import, scope: FileScope
 39) -> List[Tuple[str, str]]:
 40    """
 41    Handle the parsing of import aliases
 42
 43    Args:
 44        symbol_aliases (Dict): json dict from solc
 45        import_directive (Import): current import directive
 46        scope (FileScope): current file scape
 47
 48    Returns:
 49
 50    """
 51    aliases = []
 52    for symbol_alias in symbol_aliases:
 53        print(symbol_alias)
 54        if "foreign" in symbol_alias and "local" in symbol_alias:
 55            if isinstance(symbol_alias["foreign"], dict) and "name" in symbol_alias["foreign"]:
 56
 57                original_name = symbol_alias["foreign"]["name"]
 58                local_name = symbol_alias["local"]
 59                print(local_name)
 60                aliases.append((local_name, original_name))
 61                # import_directive.renaming[local_name] = original_name
 62                # # Assuming that two imports cannot collide in renaming
 63                # scope.renaming[local_name] = original_name
 64
 65            # This path should only be hit for the malformed AST of solc 0.5.12 where
 66            # the foreign identifier cannot be found but is required to resolve the alias.

 68            elif symbol_alias["local"]:
 69                raise SlitherException(
 70                    "Cannot resolve local alias for import directive due to malformed AST. Please upgrade to solc 0.6.0 or higher."

145    def parse_top_level_from_json(self, json_data: str) -> bool:
146        try:
147            data_loaded = json.loads(json_data)
148            # Truffle AST
149            if "ast" in data_loaded:
150                self.parse_top_level_from_loaded_json(data_loaded["ast"], data_loaded["sourcePath"])
151                return True
152            # solc AST, where the non-json text was removed
153            if "attributes" in data_loaded:
154                filename = data_loaded["attributes"]["absolutePath"]
155            else:
156                filename = data_loaded["absolutePath"]
157            self.parse_top_level_from_loaded_json(data_loaded, filename)
158            return True
159        except ValueError:
160
161            first = json_data.find("{")
162            if first != -1:
163                last = json_data.rfind("}") + 1
164                filename = json_data[0:first]
165                json_data = json_data[first:last]
166
167                data_loaded = json.loads(json_data)
168                self.parse_top_level_from_loaded_json(data_loaded, filename)
169                return True
170            return False

172    def _parse_enum(self, top_level_data: Dict, filename: str):
173        if self.is_compact_ast:
174            name = top_level_data["name"]
175            canonicalName = top_level_data["canonicalName"]
176        else:
177            name = top_level_data["attributes"][self.get_key()]
178            if "canonicalName" in top_level_data["attributes"]:
179                canonicalName = top_level_data["attributes"]["canonicalName"]
180            else:

182        values = []
183        children = (

202    def parse_top_level_from_loaded_json(
203        self, data_loaded: Dict, filename: str
204    ):  # pylint: disable=too-many-branches,too-many-statements,too-many-locals
205        if "nodeType" in data_loaded:
206            self._is_compact_ast = True
207
208        if "sourcePaths" in data_loaded:
209            for sourcePath in data_loaded["sourcePaths"]:
210                if os.path.isfile(sourcePath):
211                    self._compilation_unit.core.add_source_code(sourcePath)
212                    
213        if data_loaded[self.get_key()] == "root":
214            logger.error("solc <0.4 is not supported")
215            return
216        if data_loaded[self.get_key()] == "SourceUnit":
217            self._parse_source_unit(data_loaded, filename)
218        else:
219            logger.error("solc version is not supported")
220            return
221
222        if self.get_children() not in data_loaded:
223            return
224        scope = self.compilation_unit.get_scope(filename)
225
226        for top_level_data in data_loaded[self.get_children()]:
227            if top_level_data[self.get_key()] == "ContractDefinition":
228                contract = Contract(self._compilation_unit, scope)
229                contract_parser = ContractSolc(self, contract, top_level_data)
230                scope.contracts[contract.name] = contract
231                scope.symbol_map[contract.name + top_level_data["src"]] = DeclarationType.Contract
232                if "src" in top_level_data:
233                    contract.set_offset(top_level_data["src"], self._compilation_unit)
234
235                self._underlying_contract_to_parser[contract] = contract_parser
236
237            elif top_level_data[self.get_key()] == "PragmaDirective":
238                if self._is_compact_ast:
239                    pragma = Pragma(top_level_data["literals"], scope)
240                    scope.pragmas.add(pragma)
241                else:
242                    pragma = Pragma(top_level_data["attributes"]["literals"], scope)
243                    scope.pragmas.add(pragma)
244                pragma.set_offset(top_level_data["src"], self._compilation_unit)
245                self._compilation_unit.pragma_directives.append(pragma)
246
247            elif top_level_data[self.get_key()] == "UsingForDirective":
248                scope = self.compilation_unit.get_scope(filename)
249                usingFor = UsingForTopLevel(scope)
250                usingFor_parser = UsingForTopLevelSolc(usingFor, top_level_data, self)
251                usingFor.set_offset(top_level_data["src"], self._compilation_unit)
252                scope.using_for_directives.add(usingFor)
253
254                self._compilation_unit.using_for_top_level.append(usingFor)
255                self._using_for_top_level_parser.append(usingFor_parser)
256
257            elif top_level_data[self.get_key()] == "ImportDirective":
258                if self.is_compact_ast:
259                    import_directive = Import(
260                        Path(
261                            top_level_data["absolutePath"],
262                        ),
263                        scope,
264                    )
265                    scope.imports.add(import_directive)
266                    # TODO investigate unitAlias in version < 0.7 and legacy ast
267                    if "unitAlias" in top_level_data:
268                        import_directive.alias = top_level_data["unitAlias"]
269                    if "symbolAliases" in top_level_data:
270                        symbol_aliases = top_level_data["symbolAliases"]
271                        alias_to_orig = _handle_import_aliases(symbol_aliases, import_directive, scope)
272                else:
273                    import_directive = Import(
274                        Path(
275                            top_level_data["attributes"].get("absolutePath", ""),
276                        ),
277                        scope,
278                    )
279                    scope.imports.add(import_directive)
280                    # TODO investigate unitAlias in version < 0.7 and legacy ast
281                    if (
282                        "attributes" in top_level_data
283                        and "unitAlias" in top_level_data["attributes"]
284                    ):
285                        import_directive.alias = top_level_data["attributes"]["unitAlias"]
286                import_directive.set_offset(top_level_data["src"], self._compilation_unit)
287                self._compilation_unit.import_directives.append(import_directive)
288
289
290                #maybe we could apply the aliases here
291                # class Module:
292                # def item_import
293                # (
294                    # src: &Path,
295                    # item: &Ident,
296                    # dst: &Path,
297                    # alias: Option<Ident>,
298                # ):
299                # get_module(dst).symbol_map.get(item, None)
300                # if alias:
301                # use_aliases[alias] = item
302                # self.
303                # class Items
304                    # use_aliases: Dict[str, Ident] 
305                    # symbol_map : Dict[Ident, DeclarationType]
306                    # insert_symbol Ident, 
307                # star_import (can also have alias)
308
309
310                # unit alias import submodule as alias
311                # all declarations in submodule now have the alias
312
313                # symbol alias import {a as b} from submodule
314                # all declarations in submodule with name a now have the name b
315                imported_scope = self.compilation_unit.get_scope(import_directive.filename)
316                if symbol_aliases:
317                    for alias, orig in alias_to_orig:
318                        scope.item_import(imported_scope, alias, orig)
319                elif import_directive.alias:
320                    scope.star_import(imported_scope, alias=import_directive.alias)
321                else:
322                    scope.star_import(imported_scope, alias=None)
323
324            elif top_level_data[self.get_key()] == "StructDefinition":
325                st = StructureTopLevel(self.compilation_unit, scope)
326                st.set_offset(top_level_data["src"], self._compilation_unit)
327                st_parser = StructureTopLevelSolc(st, top_level_data, self)
328                scope.structures[st.name] = st
329                scope.symbol_map[st.name + top_level_data["src"]] = DeclarationType.StructureTopLevel
330
331                self._compilation_unit.structures_top_level.append(st)
332                self._structures_top_level_parser.append(st_parser)
333
334            elif top_level_data[self.get_key()] == "EnumDefinition":
335                # Note enum don't need a complex parser, so everything is directly done
336                self._parse_enum(top_level_data, filename)
337
338            elif top_level_data[self.get_key()] == "VariableDeclaration":
339                var = TopLevelVariable(scope)
340                var_parser = TopLevelVariableSolc(var, top_level_data, self)
341                var.set_offset(top_level_data["src"], self._compilation_unit)
342
343                self._compilation_unit.variables_top_level.append(var)
344                self._variables_top_level_parser.append(var_parser)
345                scope.variables[var.name] = var
346                scope.symbol_map[var.name + top_level_data["src"]] = DeclarationType.TopLevelVariable
347            elif top_level_data[self.get_key()] == "FunctionDefinition":
348                func = FunctionTopLevel(self._compilation_unit, scope)
349                scope.functions.add(func)
350                scope.symbol_map[func.name + top_level_data["src"]] = DeclarationType.FunctionTopLevel
351                func.set_offset(top_level_data["src"], self._compilation_unit)
352                func_parser = FunctionSolc(func, top_level_data, None, self)
353
354                self._compilation_unit.functions_top_level.append(func)
355                self._functions_top_level_parser.append(func_parser)
356                self.add_function_or_modifier_parser(func_parser)
357
358            elif top_level_data[self.get_key()] == "ErrorDefinition":
359                custom_error = CustomErrorTopLevel(self._compilation_unit, scope)
360                custom_error.set_offset(top_level_data["src"], self._compilation_unit)
361
362                custom_error_parser = CustomErrorSolc(custom_error, top_level_data, self)
363                scope.custom_errors.add(custom_error)
364                scope.symbol_map[custom_error.name + top_level_data["src"]] = DeclarationType.CustomErrorTopLevel
365                self._compilation_unit.custom_errors.append(custom_error)
366                self._custom_error_parser.append(custom_error_parser)
367
368            elif top_level_data[self.get_key()] == "UserDefinedValueTypeDefinition":
369                assert "name" in top_level_data
370                alias = top_level_data["name"]
371                assert "underlyingType" in top_level_data
372                underlying_type = top_level_data["underlyingType"]
373                assert (
374                    "nodeType" in underlying_type
375                    and underlying_type["nodeType"] == "ElementaryTypeName"
376                )
377                assert "name" in underlying_type
378
379                original_type = ElementaryType(underlying_type["name"])
380
381                user_defined_type = TypeAliasTopLevel(original_type, alias, scope)
382                user_defined_type.set_offset(top_level_data["src"], self._compilation_unit)
383                self._compilation_unit.user_defined_value_types[alias] = user_defined_type
384                scope.user_defined_types[alias] = user_defined_type
385                scope.symbol_map[alias + top_level_data["src"]] = DeclarationType.TypeAlias
386
387            else:
388                raise SlitherException(f"Top level {top_level_data[self.get_key()]} not supported")

390    def _parse_source_unit(self, data: Dict, filename: str):
391        if data[self.get_key()] != "SourceUnit":
392            return  # handle solc prior 0.3.6
393
394        # match any char for filename
395        # filename can contain space, /, -, ..
396        name_candidates = re.findall("=+ (.+) =+", filename)
397        if name_candidates:
398            assert len(name_candidates) == 1
399            name: str = name_candidates[0]
400        else:
401            name = filename
402
403        sourceUnit = -1  # handle old solc, or error
404        if "src" in data:
405            sourceUnit_candidates = re.findall("[0-9]*:[0-9]*:([0-9]*)", data["src"])
406            if len(sourceUnit_candidates) == 1:
407                sourceUnit = int(sourceUnit_candidates[0])
408        if sourceUnit == -1:
409            # if source unit is not found
410            # We can still deduce it, by assigning to the last source_code added
411            # This works only for crytic compile.
412            # which used --combined-json ast, rather than --ast-json
413            # As a result -1 is not used as index
414            if self._compilation_unit.core.crytic_compile is not None:
415                sourceUnit = len(self._compilation_unit.core.source_code)
416
417        self._compilation_unit.source_units[sourceUnit] = name
418        if os.path.isfile(name) and not name in self._compilation_unit.core.source_code:
419            self._compilation_unit.core.add_source_code(name)
420        else:
421            lib_name = os.path.join("node_modules", name)
422            if os.path.isfile(lib_name) and not name in self._compilation_unit.core.source_code:
423                self._compilation_unit.core.add_source_code(lib_name)

440    def parse_contracts(self):  # pylint: disable=too-many-statements,too-many-branches
441        if not self._underlying_contract_to_parser:
442            logger.info(
443                f"No contract were found in {self._compilation_unit.core.filename}, check the correct compilation"
444            )
445        if self._parsed:
446            raise Exception("Contract analysis can be run only once!")
447
448        # First we save all the contracts in a dict
449        # the key is the contractid
450        for contract in self._underlying_contract_to_parser:
451            if (
452                contract.name.startswith("SlitherInternalTopLevelContract")
453                and not contract.is_top_level
454            ):
455                raise SlitherException(
456                    # region multi-line-string
457                    """Your codebase has a contract named 'SlitherInternalTopLevelContract'.
458Please rename it, this name is reserved for Slither's internals"""
459                    # endregion multi-line
460                )
461            self._contracts_by_id[contract.id] = contract
462            self._compilation_unit.contracts.append(contract)
463
464        # Update of the inheritance
465        for contract_parser in self._underlying_contract_to_parser.values():
466            # remove the first elem in linearizedBaseContracts as it is the contract itself
467            ancestors = []
468            fathers = []
469            father_constructors = []
470            # try:
471            # Resolve linearized base contracts.
472            missing_inheritance = None
473
474            for i in contract_parser.linearized_base_contracts[1:]:
475                if i in contract_parser.remapping:
476                    contract_name = contract_parser.remapping[i]
477                    print(contract_name)
478                    if contract_name in contract_parser.underlying_contract.file_scope.renaming:
479                        contract_name = contract_parser.underlying_contract.file_scope.renaming[
480                            contract_name 
481                        ]
482                    target = contract_parser.underlying_contract.file_scope.get_contract_from_name(
483                        contract_name
484                    )
485                    assert target
486                    ancestors.append(target)
487                elif i in self._contracts_by_id:
488                    ancestors.append(self._contracts_by_id[i])
489                else:
490                    missing_inheritance = i
491
492            # Resolve immediate base contracts
493            for i in contract_parser.baseContracts:
494                if i in contract_parser.remapping:
495                    fathers.append(
496                        contract_parser.underlying_contract.file_scope.get_contract_from_name(
497                            contract_parser.remapping[i]
498                        )
499                        # self._compilation_unit.get_contract_from_name(contract_parser.remapping[i])
500                    )
501                elif i in self._contracts_by_id:
502                    fathers.append(self._contracts_by_id[i])
503                else:
504                    missing_inheritance = i
505
506            # Resolve immediate base constructor calls
507            for i in contract_parser.baseConstructorContractsCalled:
508                if i in contract_parser.remapping:
509                    father_constructors.append(
510                        contract_parser.underlying_contract.file_scope.get_contract_from_name(
511                            contract_parser.remapping[i]
512                        )
513                        # self._compilation_unit.get_contract_from_name(contract_parser.remapping[i])
514                    )
515                elif i in self._contracts_by_id:
516                    father_constructors.append(self._contracts_by_id[i])
517                else:
518                    missing_inheritance = i
519
520            contract_parser.underlying_contract.set_inheritance(
521                ancestors, fathers, father_constructors
522            )
523
524            if missing_inheritance:
525                self._compilation_unit.contracts_with_missing_inheritance.add(
526                    contract_parser.underlying_contract
527                )
528                txt = f"Missing inheritance {contract_parser.underlying_contract} ({contract_parser.compilation_unit.crytic_compile_compilation_unit.unique_id})\n"
529                txt += f"Missing inheritance ID: {missing_inheritance}\n"
530                if contract_parser.underlying_contract.inheritance:
531                    txt += "Inheritance found:\n"
532                    for contract_inherited in contract_parser.underlying_contract.inheritance:
533                        txt += f"\t - {contract_inherited} (ID {contract_inherited.id})\n"
534                contract_parser.log_incorrect_parsing(txt)
535
536                contract_parser.set_is_analyzed(True)
537                contract_parser.delete_content()
538
539        contracts_to_be_analyzed = list(self._underlying_contract_to_parser.values())
540
541        # Any contract can refer another contract enum without need for inheritance
542        self._analyze_all_enums(contracts_to_be_analyzed)
543        # pylint: disable=expression-not-assigned
544        [c.set_is_analyzed(False) for c in self._underlying_contract_to_parser.values()]
545
546        libraries = [
547            c for c in contracts_to_be_analyzed if c.underlying_contract.contract_kind == "library"
548        ]
549        contracts_to_be_analyzed = [
550            c for c in contracts_to_be_analyzed if c.underlying_contract.contract_kind != "library"
551        ]
552
553        # We first parse the struct/variables/functions/contract
554        self._analyze_first_part(contracts_to_be_analyzed, libraries)
555        # pylint: disable=expression-not-assigned
556        [c.set_is_analyzed(False) for c in self._underlying_contract_to_parser.values()]
557
558        # We analyze the struct and parse and analyze the events
559        # A contract can refer in the variables a struct or a event from any contract
560        # (without inheritance link)
561        self._analyze_second_part(contracts_to_be_analyzed, libraries)
562        [c.set_is_analyzed(False) for c in self._underlying_contract_to_parser.values()]
563
564        # Then we analyse state variables, functions and modifiers
565        self._analyze_third_part(contracts_to_be_analyzed, libraries)
566        [c.set_is_analyzed(False) for c in self._underlying_contract_to_parser.values()]
567
568        self._analyze_using_for(contracts_to_be_analyzed, libraries)
569
570        self._parsed = True

203    def parse_top_level_from_loaded_json(
204        self, data_loaded: Dict, filename: str
205    ):  # pylint: disable=too-many-branches,too-many-statements,too-many-locals
206        if "nodeType" in data_loaded:
207            self._is_compact_ast = True
208
209        if "sourcePaths" in data_loaded:
210            for sourcePath in data_loaded["sourcePaths"]:
211                if os.path.isfile(sourcePath):
212                    self._compilation_unit.core.add_source_code(sourcePath)
213                    
214        if data_loaded[self.get_key()] == "root":
215            logger.error("solc <0.4 is not supported")
216            return
217        if data_loaded[self.get_key()] == "SourceUnit":
218            self._parse_source_unit(data_loaded, filename)
219        else:
220            logger.error("solc version is not supported")
221            return
222
223        if self.get_children() not in data_loaded:
224            return
225        scope = self.compilation_unit.get_scope(filename)
226
227        for top_level_data in data_loaded[self.get_children()]:
228            if top_level_data[self.get_key()] == "ContractDefinition":
229                contract = Contract(self._compilation_unit, scope)
230                contract_parser = ContractSolc(self, contract, top_level_data)
231                scope.contracts[contract.name] = contract
232                scope.symbol_map[contract.name + top_level_data["src"]] = DeclarationType.Contract
233                if "src" in top_level_data:
234                    contract.set_offset(top_level_data["src"], self._compilation_unit)
235
236                self._underlying_contract_to_parser[contract] = contract_parser
237
238            elif top_level_data[self.get_key()] == "PragmaDirective":
239                if self._is_compact_ast:
240                    pragma = Pragma(top_level_data["literals"], scope)
241                    scope.pragmas.add(pragma)
242                else:
243                    pragma = Pragma(top_level_data["attributes"]["literals"], scope)
244                    scope.pragmas.add(pragma)
245                pragma.set_offset(top_level_data["src"], self._compilation_unit)
246                self._compilation_unit.pragma_directives.append(pragma)
247
248            elif top_level_data[self.get_key()] == "UsingForDirective":
249                scope = self.compilation_unit.get_scope(filename)
250                usingFor = UsingForTopLevel(scope)
251                usingFor_parser = UsingForTopLevelSolc(usingFor, top_level_data, self)
252                usingFor.set_offset(top_level_data["src"], self._compilation_unit)
253                scope.using_for_directives.add(usingFor)
254
255                self._compilation_unit.using_for_top_level.append(usingFor)
256                self._using_for_top_level_parser.append(usingFor_parser)
257
258            elif top_level_data[self.get_key()] == "ImportDirective":
259                if self.is_compact_ast:
260                    import_directive = Import(
261                        Path(
262                            top_level_data["absolutePath"],
263                        ),
264                        scope,
265                    )
266                    scope.imports.add(import_directive)
267                    # TODO investigate unitAlias in version < 0.7 and legacy ast
268                    if "unitAlias" in top_level_data:
269                        import_directive.alias = top_level_data["unitAlias"]
270                    if "symbolAliases" in top_level_data:
271                        symbol_aliases = top_level_data["symbolAliases"]
272                        alias_to_orig = _handle_import_aliases(symbol_aliases, import_directive, scope)
273                else:
274                    import_directive = Import(
275                        Path(
276                            top_level_data["attributes"].get("absolutePath", ""),
277                        ),
278                        scope,
279                    )
280                    scope.imports.add(import_directive)
281                    # TODO investigate unitAlias in version < 0.7 and legacy ast
282                    if (
283                        "attributes" in top_level_data
284                        and "unitAlias" in top_level_data["attributes"]
285                    ):
286                        import_directive.alias = top_level_data["attributes"]["unitAlias"]
287                import_directive.set_offset(top_level_data["src"], self._compilation_unit)
288                self._compilation_unit.import_directives.append(import_directive)
289
290
291                #maybe we could apply the aliases here
292                # class Module:
293                # def item_import
294                # (
295                    # src: &Path,
296                    # item: &Ident,
297                    # dst: &Path,
298                    # alias: Option<Ident>,
299                # ):
300                # get_module(dst).symbol_map.get(item, None)
301                # if alias:
302                # use_aliases[alias] = item
303                # self.
304                # class Items
305                    # use_aliases: Dict[str, Ident] 
306                    # symbol_map : Dict[Ident, DeclarationType]
307                    # insert_symbol Ident, 
308                # star_import (can also have alias)
309
310
311                # unit alias import submodule as alias
312                # all declarations in submodule now have the alias
313
314                # symbol alias import {a as b} from submodule
315                # all declarations in submodule with name a now have the name b
316                imported_scope = self.compilation_unit.get_scope(import_directive.filename)
317                if symbol_aliases:
318                    for alias, orig in alias_to_orig:
319                        scope.item_import(imported_scope, alias, orig)
320                elif import_directive.alias:
321                    scope.star_import(imported_scope, alias=import_directive.alias)
322                else:
323                    scope.star_import(imported_scope, alias=None)
324
325            elif top_level_data[self.get_key()] == "StructDefinition":
326                st = StructureTopLevel(self.compilation_unit, scope)
327                st.set_offset(top_level_data["src"], self._compilation_unit)
328                st_parser = StructureTopLevelSolc(st, top_level_data, self)
329                scope.structures[st.name] = st
330                scope.symbol_map[st.name + top_level_data["src"]] = DeclarationType.StructureTopLevel
331
332                self._compilation_unit.structures_top_level.append(st)
333                self._structures_top_level_parser.append(st_parser)

335            elif top_level_data[self.get_key()] == "EnumDefinition":

339            elif top_level_data[self.get_key()] == "VariableDeclaration":

342                var.set_offset(top_level_data["src"], self._compilation_unit)
343
344                self._compilation_unit.variables_top_level.append(var)
345                self._variables_top_level_parser.append(var_parser)
346                scope.variables[var.name] = var
347                scope.symbol_map[var.name + top_level_data["src"]] = DeclarationType.TopLevelVariable
348            elif top_level_data[self.get_key()] == "FunctionDefinition":
349                func = FunctionTopLevel(self._compilation_unit, scope)
350                scope.functions.add(func)
351                scope.symbol_map[func.name + top_level_data["src"]] = DeclarationType.FunctionTopLevel
352                func.set_offset(top_level_data["src"], self._compilation_unit)
353                func_parser = FunctionSolc(func, top_level_data, None, self)
354
355                self._compilation_unit.functions_top_level.append(func)
356                self._functions_top_level_parser.append(func_parser)
357                self.add_function_or_modifier_parser(func_parser)
358
359            elif top_level_data[self.get_key()] == "ErrorDefinition":
360                custom_error = CustomErrorTopLevel(self._compilation_unit, scope)
361                custom_error.set_offset(top_level_data["src"], self._compilation_unit)
362
363                custom_error_parser = CustomErrorSolc(custom_error, top_level_data, self)
364                scope.custom_errors.add(custom_error)
365                scope.symbol_map[custom_error.name + top_level_data["src"]] = DeclarationType.CustomErrorTopLevel
366                self._compilation_unit.custom_errors.append(custom_error)
367                self._custom_error_parser.append(custom_error_parser)

369            elif top_level_data[self.get_key()] == "UserDefinedValueTypeDefinition":

373                underlying_type = top_level_data["underlyingType"]
374                assert (
375                    "nodeType" in underlying_type
376                    and underlying_type["nodeType"] == "ElementaryTypeName"
377                )
378                assert "name" in underlying_type

380                original_type = ElementaryType(underlying_type["name"])

383                user_defined_type.set_offset(top_level_data["src"], self._compilation_unit)
384                self._compilation_unit.user_defined_value_types[alias] = user_defined_type
385                scope.user_defined_types[alias] = user_defined_type
386                scope.symbol_map[alias + top_level_data["src"]] = DeclarationType.TypeAlias
387
388            else:
389                raise SlitherException(f"Top level {top_level_data[self.get_key()]} not supported")

391    def _parse_source_unit(self, data: Dict, filename: str):
392        if data[self.get_key()] != "SourceUnit":
393            return  # handle solc prior 0.3.6
394
395        # match any char for filename
396        # filename can contain space, /, -, ..
397        name_candidates = re.findall("=+ (.+) =+", filename)
398        if name_candidates:
399            assert len(name_candidates) == 1
400            name: str = name_candidates[0]
401        else:
402            name = filename
403
404        sourceUnit = -1  # handle old solc, or error
405        if "src" in data:
406            sourceUnit_candidates = re.findall("[0-9]*:[0-9]*:([0-9]*)", data["src"])
407            if len(sourceUnit_candidates) == 1:
408                sourceUnit = int(sourceUnit_candidates[0])
409        if sourceUnit == -1:
410            # if source unit is not found
411            # We can still deduce it, by assigning to the last source_code added
412            # This works only for crytic compile.
413            # which used --combined-json ast, rather than --ast-json
414            # As a result -1 is not used as index
415            if self._compilation_unit.core.crytic_compile is not None:
416                sourceUnit = len(self._compilation_unit.core.source_code)
417
418        self._compilation_unit.source_units[sourceUnit] = name
419        if os.path.isfile(name) and not name in self._compilation_unit.core.source_code:
420            self._compilation_unit.core.add_source_code(name)
421        else:
422            lib_name = os.path.join("node_modules", name)
423            if os.path.isfile(lib_name) and not name in self._compilation_unit.core.source_code:
424                self._compilation_unit.core.add_source_code(lib_name)
