 56        assert self._unparsed_expression is None
 57        self._unparsed_expression = expression

 59    def analyze_expressions(self) -> None:
 60        if self._node.type == NodeType.VARIABLE and not self._node.expression:
 61            expression = self._node.variable_declaration.expression
 62            if expression:
 63                self._node.add_expression(expression)
 64        if self._unparsed_expression:
 65            expression = parse_yul(self._scope, self, self._unparsed_expression)
 66            if expression:
 67                self._node.add_expression(expression)
 68
 69        if self._node.expression:
 70            if self._node.type == NodeType.VARIABLE:
 71                # Update the expression to be an assignement to the variable
 72                variable_declaration = self._node.variable_declaration
 73                if variable_declaration:
 74                    _expression = AssignmentOperation(
 75                        Identifier(self._node.variable_declaration),
 76                        self._node.expression,
 77                        AssignmentOperationType.ASSIGN,
 78                        variable_declaration.type,
 79                    )
 80                    _expression.set_offset(
 81                        self._node.expression.source_mapping, self._node.compilation_unit
 82                    )
 83                    self._node.add_expression(_expression, bypass_verif_empty=True)
 84
 85            expression = self._node.expression
 86            read_var = ReadVar(expression)
 87            self._node.variables_read_as_expression = read_var.result()
 88
 89            write_var = WriteVar(expression)
 90            self._node.variables_written_as_expression = write_var.result()
 91
 92            find_call = FindCalls(expression)
 93            self._node.calls_as_expression = find_call.result()
 94            self._node.external_calls_as_expressions = [
 95                c for c in self._node.calls_as_expression if not isinstance(c.called, Identifier)
 96            ]
 97            self._node.internal_calls_as_expressions = [
 98                c for c in self._node.calls_as_expression if isinstance(c.called, Identifier)
 99            ]

106def _name_to_yul_name(variable_name: str, yul_id: List[str]) -> str:
107    """
108    Translate the variable name to a unique yul name
109    Within the same function, yul blocks can declare
110    different variables with the same name
111    We need to create unique name per variable
112    to prevent collision during the SSA generation
113
114    :param var:
115    :param yul_id:
116    :return:
117    """
118    return variable_name + f"_{'_'.join(yul_id)}"

213class YulFunction(YulScope):
214    __slots__ = ["_function", "_root", "_ast", "_nodes", "_entrypoint", "node_scope"]
215
216    def __init__(
217        self, func: Function, root: YulScope, ast: Dict, node_scope: Union[Function, Scope]
218    ):
219        super().__init__(root.contract, root.id + [ast["name"]], parent_func=root.parent_func)
220
221        assert ast["nodeType"] == "YulFunctionDefinition"
222
223        self._function: Function = func
224        self._root: YulScope = root
225        self._ast: Dict = ast
226
227        # start initializing the underlying function
228
229        func.name = ast["name"]
230        func.set_visibility("private")
231        if isinstance(func, SourceMapping):
232            func.set_offset(ast["src"], root.compilation_unit)
233        if isinstance(func, FunctionContract):
234            func.set_contract(root.contract)
235            func.set_contract_declarer(root.contract)
236        func.compilation_unit = root.compilation_unit
237        func.internal_scope = root.id
238        func.is_implemented = True
239        self.node_scope = node_scope
240
241        self._nodes: List[YulNode] = []
242        self._entrypoint = self.new_node(NodeType.ASSEMBLY, ast["src"])
243        func.entry_point = self._entrypoint.underlying_node
244
245        self.add_yul_local_function(self)
246
247    @property
248    def underlying(self) -> Function:
249        return self._function
250
251    @property
252    def function(self) -> Function:
253        return self._function
254
255    def convert_body(self) -> None:
256        node = self.new_node(NodeType.ENTRYPOINT, self._ast["src"])
257        link_underlying_nodes(self._entrypoint, node)
258
259        for param in self._ast.get("parameters", []):
260            node = convert_yul(self, node, param, self.node_scope)
261            self._function.add_parameters(
262                self.get_yul_local_variable_from_name(param["name"]).underlying
263            )
264
265        for ret in self._ast.get("returnVariables", []):
266            node = convert_yul(self, node, ret, self.node_scope)
267            self._function.add_return(self.get_yul_local_variable_from_name(ret["name"]).underlying)
268
269        convert_yul(self, node, self._ast["body"], self.node_scope)
270
271    def parse_body(self) -> None:
272        for node in self._nodes:
273            node.analyze_expressions()
274
275    def new_node(self, node_type, src) -> YulNode:
276        if self._function:
277            node = self._function.new_node(node_type, src, self.node_scope)
278        else:
279            raise SlitherException("standalone yul objects are not supported yet")
280
281        yul_node = YulNode(node, self)
282        self._nodes.append(yul_node)
283        return yul_node

342# The functions in this region, at a high level, will extract the control flow
343# structures and metadata from the input AST. These include things like function
344# definitions and local variables.
345#
346# Each function takes three parameters:
347#     1)  root is the current YulScope, where you can find things like local variables
348#     2)  parent is the previous YulNode, which you'll have to link to
349#     3)  ast is a dictionary and is the current node in the Yul ast being converted
350#
351# Each function must return a single parameter:
352#     1)  the new YulNode that the CFG ends at
353#
354# The entrypoint is the function at the end of this region, `convert_yul`, which
355# dispatches to a specialized function based on a lookup dictionary.
356

366def convert_yul_function_definition(
367    root: YulScope, parent: YulNode, ast: Dict, node_scope: Union[Function, Scope]
368) -> YulNode:
369    top_node_scope = node_scope
370    while not isinstance(top_node_scope, Function):
371        top_node_scope = top_node_scope.father
372
373    func: Union[FunctionTopLevel, FunctionContract]
374    if isinstance(top_node_scope, FunctionTopLevel):
375        scope = root.file_scope
376        func = FunctionTopLevel(root.compilation_unit, scope)
377        # Note: we do not add the function in the scope
378        # While its a top level function, it is not accessible outside of the function definition
379        # In practice we should probably have a specific function type for function defined within a function
380    else:
381        func = FunctionContract(root.compilation_unit)
382
383    func.function_language = FunctionLanguage.Yul
384    yul_function = YulFunction(func, root, ast, node_scope)
385
386    if root.contract:
387        root.contract.add_function(func)
388
389    root.compilation_unit.add_function(func)
390    root.add_yul_local_function(yul_function)
391
392    yul_function.convert_body()
393    yul_function.parse_body()
394
395    return parent

433def convert_yul_if(
434    root: YulScope, parent: YulNode, ast: Dict, node_scope: Union[Function, Scope]
435) -> YulNode:
436    # we're cheating and pretending that yul supports if/else so we can convert switch cleaner
437
438    src = ast["src"]
439    condition_ast = ast["condition"]
440    true_body_ast = ast["body"]
441    false_body_ast = ast["false_body"] if "false_body" in ast else None
442
443    condition = root.new_node(NodeType.IF, src)
444    end = root.new_node(NodeType.ENDIF, src)
445
446    condition.add_unparsed_expression(condition_ast)
447
448    true_body = convert_yul(root, condition, true_body_ast, node_scope)
449
450    if false_body_ast:
451        false_body = convert_yul(root, condition, false_body_ast, node_scope)
452        link_underlying_nodes(false_body, end)
453    else:
454        link_underlying_nodes(condition, end)
455
456    link_underlying_nodes(parent, condition)
457    link_underlying_nodes(true_body, end)
458
459    return end

462def convert_yul_switch(
463    root: YulScope, parent: YulNode, ast: Dict, node_scope: Union[Function, Scope]
464) -> YulNode:
465    """
466    This is unfortunate. We don't really want a switch in our IR so we're going to
467    translate it into a series of if/else statements.
468    """
469    cases_ast = ast["cases"]
470    expression_ast = ast["expression"]
471
472    # this variable stores the result of the expression so we don't accidentally compute it more than once
473    switch_expr_var = f"switch_expr_{ast['src'].replace(':', '_')}"
474
475    rewritten_switch = {
476        "nodeType": "YulBlock",
477        "src": ast["src"],
478        "statements": [
479            {
480                "nodeType": "YulVariableDeclaration",
481                "src": expression_ast["src"],
482                "variables": [
483                    {
484                        "nodeType": "YulTypedName",
485                        "src": expression_ast["src"],
486                        "name": switch_expr_var,
487                        "type": "",
488                    },
489                ],
490                "value": expression_ast,
491            },
492        ],
493    }
494
495    last_if: Optional[Dict] = None
496
497    default_ast = None
498
499    for case_ast in cases_ast:
500        body_ast = case_ast["body"]
501        value_ast = case_ast["value"]
502
503        if value_ast == "default":
504            default_ast = case_ast
505            continue
506
507        current_if = {
508            "nodeType": "YulIf",
509            "src": case_ast["src"],
510            "condition": {
511                "nodeType": "YulFunctionCall",
512                "src": case_ast["src"],
513                "functionName": {
514                    "nodeType": "YulIdentifier",
515                    "src": case_ast["src"],
516                    "name": "eq",
517                },
518                "arguments": [
519                    {
520                        "nodeType": "YulIdentifier",
521                        "src": case_ast["src"],
522                        "name": switch_expr_var,
523                    },
524                    value_ast,
525                ],
526            },
527            "body": body_ast,
528        }
529
530        if last_if:
531            last_if["false_body"] = current_if  # pylint: disable=unsupported-assignment-operation
532        else:
533            rewritten_switch["statements"].append(current_if)
534
535        last_if = current_if
536
537    if default_ast:
538        body_ast = default_ast["body"]
539
540        if last_if:
541            last_if["false_body"] = body_ast
542        else:
543            rewritten_switch["statements"].append(body_ast)
544
545    return convert_yul(root, parent, rewritten_switch, node_scope)

548def convert_yul_for_loop(
549    root: YulScope, parent: YulNode, ast: Dict, node_scope: Union[Function, Scope]
550) -> YulNode:
551    pre_ast = ast["pre"]
552    condition_ast = ast["condition"]
553    post_ast = ast["post"]
554    body_ast = ast["body"]
555
556    start_loop = root.new_node(NodeType.STARTLOOP, ast["src"])
557    end_loop = root.new_node(NodeType.ENDLOOP, ast["src"])
558
559    link_underlying_nodes(parent, start_loop)
560
561    pre = convert_yul(root, start_loop, pre_ast, node_scope)
562
563    condition = root.new_node(NodeType.IFLOOP, condition_ast["src"])
564    condition.add_unparsed_expression(condition_ast)
565    link_underlying_nodes(pre, condition)
566
567    link_underlying_nodes(condition, end_loop)
568
569    body = convert_yul(root, condition, body_ast, node_scope)
570
571    post = convert_yul(root, body, post_ast, node_scope)
572
573    link_underlying_nodes(post, condition)
574
575    return end_loop

701def parse_yul_function_call(root: YulScope, node: YulNode, ast: Dict) -> Optional[Expression]:
702    args = [parse_yul(root, node, arg) for arg in ast["arguments"]]
703    ident = parse_yul(root, node, ast["functionName"])
704
705    if not isinstance(ident, Identifier):
706        raise SlitherException("expected identifier from parsing function name")
707
708    if isinstance(ident.value, YulBuiltin):
709        name = ident.value.name
710        if name in binary_ops:
711            if name in ["shl", "shr", "sar"]:
712                # lmao ok
713                return BinaryOperation(args[1], args[0], binary_ops[name])
714
715            return BinaryOperation(args[0], args[1], binary_ops[name])
716
717        if name in unary_ops:
718            return UnaryOperation(args[0], unary_ops[name])
719
720        if name == "stop":
721            name = "return"
722            ident = Identifier(SolidityFunction(format_function_descriptor(name)))
723            args = [
724                Literal("0", ElementaryType("uint256")),
725                Literal("0", ElementaryType("uint256")),
726            ]
727
728        else:
729            ident = Identifier(SolidityFunction(format_function_descriptor(ident.value.name)))
730
731    if isinstance(ident.value, Function):
732        return CallExpression(ident, args, vars_to_typestr(ident.value.returns))
733    if isinstance(ident.value, SolidityFunction):
734        return CallExpression(ident, args, vars_to_typestr(ident.value.return_type))
735
736    raise SlitherException(f"unexpected function call target type {str(type(ident.value))}")

748def _parse_yul_magic_suffixes(name: str, root: YulScope) -> Optional[Expression]:
749    # check for magic suffixes
750    # TODO: the following leads to wrong IR
751    # Currently SlithIR doesnt support raw access to memory
752    # So things like .offset/.slot will return the variable
753    # Instaed of the actual offset/slot
754    if name.endswith(("_slot", ".slot")):
755        potential_name = name[:-5]
756        variable_found = _check_for_state_variable_name(root, potential_name)
757        if variable_found:
758            return variable_found
759        var = root.function.get_local_variable_from_name(potential_name)
760        if var and var.is_storage:
761            return Identifier(var)
762    if name.endswith(("_offset", ".offset")):
763        potential_name = name[:-7]
764        variable_found = _check_for_state_variable_name(root, potential_name)
765        if variable_found:
766            return variable_found
767        var = root.function.get_local_variable_from_name(potential_name)
768        if var and var.location == "calldata":
769            return Identifier(var)
770    if name.endswith(".length"):
771        # TODO: length should create a new IP operation LENGTH var
772        # The code below is an hotfix to allow slither to process length in yul
773        # Until we have a better support
774        potential_name = name[:-7]
775        var = root.function.get_local_variable_from_name(potential_name)
776        if var and var.location == "calldata":
777            return Identifier(var)
778    return None

781def parse_yul_identifier(root: YulScope, _node: YulNode, ast: Dict) -> Optional[Expression]:
782    name = ast["name"]
783
784    if name in builtins:
785        return Identifier(YulBuiltin(name))
786
787    # check function-scoped variables
788    parent_func = root.parent_func
789    if parent_func:
790        local_variable = parent_func.get_local_variable_from_name(name)
791        if local_variable:
792            return Identifier(local_variable)
793
794        if isinstance(parent_func, FunctionContract):
795            assert parent_func.contract
796            state_variable = parent_func.contract.get_state_variable_from_name(name)
797            if state_variable:
798                return Identifier(state_variable)
799
800    # check yul-scoped variable
801    variable = root.get_yul_local_variable_from_name(name)
802    if variable:
803        return Identifier(variable.underlying)
804
805    # check yul-scoped function
806
807    func = root.get_yul_local_function_from_name(name)
808    if func:
809        return Identifier(func.underlying)
810
811    magic_suffix = _parse_yul_magic_suffixes(name, root)
812    if magic_suffix:
813        return magic_suffix
814
815    ret, _ = find_top_level(name, root.file_scope)
816    if ret:
817        return Identifier(ret)
818
819    raise SlitherException(f"unresolved reference to identifier {name}")

822def parse_yul_literal(_root: YulScope, _node: YulNode, ast: Dict) -> Optional[Expression]:
823    kind = ast["kind"]
824
825    if kind == "string":
826        # Solc 0.8.0 use value, 0.8.16 use hexValue - not sure when this changed was made
827        if "value" in ast:
828            value = ast["value"]
829        else:
830            value = ast["hexValue"]
831    else:
832        # number/bool
833        value = ast["value"]
834
835    if not kind:
836        kind = "bool" if value in ["true", "false"] else "uint256"
837
838    if kind == "number":
839        kind = "uint256"
840
841    return Literal(value, ElementaryType(kind))

878def vars_to_typestr(rets: List[Expression]) -> str:
879    if len(rets) == 0:
880        return ""
881    if len(rets) == 1:
882        return str(rets[0].type)
883    return f"tuple({','.join(str(ret.type) for ret in rets)})"
